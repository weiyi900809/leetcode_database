102. Binary Tree Level Order Traversal	


/**
 * Definition for a binary tree node.
 * struct TreeNode {
 * int val;
 * TreeNode *left;
 * TreeNode *right;
 * TreeNode() : val(0), left(nullptr), right(nullptr) {}
 * TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 * TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> ans;
        
        // ==========================================
        // 🚨 坑點 1：忘記防禦空樹 (Null Pointer Exception)
        // ==========================================
        // 如果輸入的樹是空的，直接把 nullptr 塞進 Queue 裡，
        // 後面呼叫 now->val 會直接引發 Segmentation Fault (死當)。
        if (root == nullptr) return ans;

        // 建立 BFS 的核心兵器：佇列 (先進先出)
        queue<TreeNode*> q;

        // 第一步：把樹根排進隊伍裡
        q.push(root);

        // 只要隊伍裡還有人，就繼續處理
        while (!q.empty()) {
            
            // ==========================================
            // 🚨 坑點 2：動態變化的 Queue Size 地雷
            // ==========================================
            // 這裡「必須」把 q.size() 存進一個固定的變數 level_size。
            // 很多初學者會寫成：for (int i = 0; i < q.size(); i++)
            // 這是致命錯誤！因為在迴圈裡面我們會不斷 push 新的小孩進去，
            // q.size() 會越變越大，導致這個迴圈永遠停不下來，無法正確「分層」。
            int level_size = q.size();
            
            // 準備一個陣列，用來裝「目前這一層」所有節點的值
            vector<int> curr_level;

            // 嚴格管控額度：只處理 level_size 這麼多個人
            for (int i = 0; i < level_size; i++) {
                
                // ==========================================
                // 🚨 坑點 3：C++ Queue 的兩段式取值
                // ==========================================
                // 在 Java 或 Python，pop() 拿掉元素的同時會順便回傳那個元素。
                // 但 C++ 設計得很嚴格：front() 只負責「看」最前面是誰，pop() 只負責「踢掉」不回傳。
                // 所以一定要先 front() 把人接住，再 pop() 踢掉，順序絕對不能錯！
                TreeNode* now = q.front();
                q.pop();
                
                // 把這個人的值存進這一層的陣列裡
                curr_level.push_back(now->val);
                
                // 如果有左小孩，排進隊伍的最後面
                if (now->left != nullptr) q.push(now->left);
                
                // 如果有右小孩，排進隊伍的最後面
                if (now->right != nullptr) q.push(now->right);
            }
            
            // 這層的額度用完了，把這一層的陣列打包，放進最終答案裡
            ans.push_back(curr_level);
        }
        
        return ans;
    }
};
