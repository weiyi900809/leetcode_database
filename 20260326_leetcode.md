104. Maximum Depth of Binary Tree

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 * int val;
 * TreeNode *left;
 * TreeNode *right;
 * TreeNode() : val(0), left(nullptr), right(nullptr) {}
 * TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 * TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int maxDepth(TreeNode* root) {
        // ==========================================
        // 1. 嚴謹的邊界防禦 (Base Case / 終止條件)
        // ==========================================
        // 當遞迴往下找到盡頭，踩到空指標 (nullptr) 時，代表這是一塊沒有房子的空地。
        // 這個「不存在的下屬」必須回報深度為 0，這也是整個遞迴能夠煞車、開始往上回傳的關鍵。
        if (root == nullptr) {
            return 0;
        }

        // ==========================================
        // 2. 收集情報 (Recursive Calls / 進入下一層)
        // ==========================================
        // 身為當前節點 (長官)，分別打電話給左手邊和右手邊的直屬下屬。
        // 要求他們去調查各自部門的最大深度，並將結果存進變數中。
        // (程式走到這裡會先暫停，一直往下鑽，直到最底層拿到 0 才會一層層算回來)
        int left_depth = maxDepth(root->left);
        int right_depth = maxDepth(root->right);

        // ==========================================
        // 3. 統整並回報 (Return Result / 結算當前層)
        // ==========================================
        // 比較左部門和右部門回報的深度，挑出數字比較大 (比較深) 的那一邊。
        // 然後加上「自己這一層」的深度 (+1)，作為最終結果回報給上一級長官。
        return (left_depth > right_depth) ? left_depth + 1 : right_depth + 1;
    }
};
