1971. Find if Path Exists in Graph





class Solution {
private:
    // ==========================================
    // DFS 遞迴偵察兵 (負責在圖中找路)
    // ==========================================
    bool dfs(int curr, int destination, vector<vector<int>>& graph, vector<bool>& visited) {
        
        // 1. 終止條件：如果現在所在的城市 curr 就是目的地 destination
        // -> 回傳 true (找到了！)
        if (curr == destination) return true;
        
        // 2. 狀態更新：在簽到表上把 curr 標記為 true (代表來過了)
        visited[curr] = true;
        
        // 3. 展開平行宇宙：翻開地圖，查看 curr 可以通往的所有鄰居
        // (優化：使用範圍 for 迴圈，直接把 graph[curr] 裡面的數字一個個拿出來當作 neighbor)
        for (int neighbor : graph[curr]) {
            
            // 如果這個鄰居還沒去過
            if (!visited[neighbor]) {
                // (優化：直接將 DFS 遞迴結果寫在 if 判斷式中)
                // 派出新的 dfs 去探索，只要有任何一條路徑回傳了 true，就立刻向上一層回傳 true！
                if (dfs(neighbor, destination, graph, visited)) {
                    return true;
                }
            }
        }

        // 4. 如果四周的鄰居都走不通，或是都去過了，這條路就是死路
        return false; 
    }

public:
    bool validPath(int n, vector<vector<int>>& edges, int source, int destination) {
        
        // 階段 1：建立地圖 (完美套用 const & 零成本讀取防禦寫法)
        vector<vector<int>> graph(n);
        for (const vector<int>& edge : edges) {
            graph[edge[0]].push_back(edge[1]);
            graph[edge[1]].push_back(edge[0]);
        }
        
        // 階段 2：建立簽到表 (大小為 n，初始狀態全為 false)
        vector<bool> visited(n, false);
        
        // 階段 3：從 source 出發，呼叫 DFS 偵察兵並回傳結果
        return dfs(source, destination, graph, visited);
    }
};
