572. Subtree of Another Tree

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 * int val;
 * TreeNode *left;
 * TreeNode *right;
 * TreeNode() : val(0), left(nullptr), right(nullptr) {}
 * TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 * TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    // ==========================================
    // 輔助函式 (特種驗證部隊)：嚴格比對兩棵樹是否「完全相同」
    // ==========================================
    bool isSameTree(TreeNode* p, TreeNode* q){
        // 1. 雙方同時到底：代表路徑上的節點都長一樣，驗證通過
        if(p == nullptr && q == nullptr) return true;
        
        // 2. 結構不對稱：其中一方先到底，代表樹的形狀不同，直接淘汰
        else if(p == nullptr || q == nullptr) return false;

        // 3. 數值比對：形狀就算一樣，值不同也是淘汰
        if(p->val != q->val) return false;

        // ==========================================
        // 🚨 坑點 1：驗證必須嚴格使用 AND (&&)
        // ==========================================
        // 兩棵樹要「一模一樣」，必須是左子樹相同「且」右子樹相同。
        // 這裡絕對不能寫成 ||，否則只要一邊長得一樣就會誤判放行。
        // (同時也要注意參數的對應：p 的左邊配 q 的左邊)
        return isSameTree(p->left, q->left) && isSameTree(p->right, q->right);
    }
    
    // ==========================================
    // 主函式 (全域搜索大隊)：在 root 森林中尋找 subRoot
    // ==========================================
    bool isSubtree(TreeNode* root, TreeNode* subRoot) {
        // ==========================================
        // 🚨 坑點 2：防呆邊界的先後順序
        // ==========================================
        // 這裡的邏輯非常嚴謹。必須「先」判斷兩者皆為空，
        //「再」判斷其中一者為空。如果反過來寫，會把皆為空的情況誤判成 false。
        // (註：LeetCode 題目有保證 subRoot 不為空，所以其實只寫 if(root == nullptr) return false; 也會過，但你這種寫法在工程上更安全且無懈可擊)
        if(root == nullptr && subRoot == nullptr) return true;
        else if(root == nullptr || subRoot == nullptr) return false;

        // ==========================================
        // 🚨 坑點 3：搜索必須寬鬆使用 OR (||)
        // ==========================================
        // 在廣大的森林裡找東西，我們不要求每一條路都有寶藏。
        // 只要符合以下「任何一種」情況就算成功找到：
        // 1. 寶藏就在「現在踩著的節點」底下 (交給 isSameTree 驗證)
        // 2. 寶藏藏在「左邊的森林」裡
        // 3. 寶藏藏在「右邊的森林」裡
        return isSameTree(root, subRoot) || 
               isSubtree(root->left, subRoot) ||  
               isSubtree(root->right, subRoot);
    }
};
