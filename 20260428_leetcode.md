643. Maximum Average Subarray I

class Solution {
public:
    double findMaxAverage(vector<int>& nums, int k) {
        
        // ==========================================
        // 階段 1：建立初始窗口 (Initial Window)
        // ==========================================
        int CurrentSum = 0;
        
        // 先把陣列最前面的 k 個數字加起來，當作我們第一個窗口的總和
        for(int i = 0; i < k; i++){
            CurrentSum += nums[i];
        }
        
        // 🚨 致命陷阱防禦：MaxSum 絕對不能初始化為 0！
        // 因為陣列裡可能全是負數 (例如 [-1, -2, -3])，如果設為 0 會回傳錯誤的答案。
        // 直接把第一個窗口的總和，當作目前歷史上的最大值。
        int MaxSum = CurrentSum;
        
        // ==========================================
        // 階段 2：開始滑動窗口 (Sliding Window)
        // ==========================================
        // 框框開始往右滑動，每次只動一格。
        // i 代表的是「剛剛滑進框框裡的新數字」的 Index。
        for(int i = k; i < nums.size(); i++){
            
            // 💡 滑動核心魔術：O(1) 時間算出新總和
            // 不需要重新跑迴圈加總！
            // 新總和 = 原本的總和 + 滑進來的新數字 (nums[i]) - 滑出去的舊數字 (nums[i-k])
            CurrentSum += nums[i] - nums[i-k];
            
            // 比較「剛剛滑出來的新總和」與「歷史最大總和」，留下比較大的那一個
            MaxSum = max(MaxSum, CurrentSum);
        }
        
        // ==========================================
        // 階段 3：結算輸出
        // ==========================================
        // 題目要求回傳平均值 (小數)。
        // 為了避免整數除法把小數點吃掉，必須加上 (double) 強制轉型。
        return (double) MaxSum / k;
    }
};
