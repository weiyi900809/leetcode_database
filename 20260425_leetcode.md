746. Min Cost Climbing Stairs

class Solution {
public:
    int minCostClimbingStairs(vector<int>& cost) {
        
        // ==========================================
        // 📍 初始狀態 (Base Cases)
        // ==========================================
        // 題目規定：你可以選擇從第 0 階或第 1 階「免費起步」。
        // 這代表在踏上第 0 階和第 1 階之前，我們口袋裡噴掉的錢都是 0。
        int prev2 = 0; // 抵達 i-2 階的累積最小花費
        int prev1 = 0; // 抵達 i-1 階的累積最小花費
        
        // ==========================================
        // 🚀 DP 狀態轉移 (開始爬樓梯)
        // ==========================================
        // 注意邊界：目標是「頂樓」，而頂樓的位子剛好在陣列的最後面 (index 為 cost.size())
        // 所以迴圈必須包含等於 (i <= cost.size())，這樣才會真的踏上頂樓！
        for (int i = 2; i <= cost.size(); i++) {
            
            // 💡 核心比價邏輯：在兩條路徑中挑「最便宜」的
            // 路線 A：從前一階 (i-1) 跨 1 步上來。
            //        代價 = 抵達前一階的累積花費 (prev1) + 離開前一階要付的過路費 (cost[i-1])
            // 路線 B：從前兩階 (i-2) 跨 2 步直接上來。
            //        代價 = 抵達前兩階的累積花費 (prev2) + 離開前兩階要付的過路費 (cost[i-2])
            int current = min(prev1 + cost[i-1], prev2 + cost[i-2]);
            
            // 🏃‍♂️ 交接棒時間 (狀態壓縮，大家往前走一格，準備算下一階)
            prev2 = prev1;   // 原本的 i-1 變成下一步的 i-2
            prev1 = current; // 剛剛算出的最新答案，變成下一步的 i-1
        }
        
        // 迴圈結束時，我們剛好抵達頂樓，此時 prev1 握著的就是整趟旅程的最少花費！
        return prev1; 
    }
};
