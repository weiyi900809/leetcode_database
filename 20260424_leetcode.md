448. Find All Numbers Disappeared in an Array

class Solution {
public:
    vector<int> findDisappearedNumbers(vector<int>& nums) {
        vector<int> result;
        
        // ==========================================
        // 階段 1：原地做記號 (In-place Modification)
        // 核心精神：把「數值」當作「Index」，去對應的抽屜打上負號
        // ==========================================
        for(int i = 0; i < nums.size(); i++){
            
            // 🚨 坑 1：越界當機 (Segmentation Fault) 預防
            // 因為前面的數字可能已經把現在這個數字變成負的了，
            // 所以要當作 Index 找抽屜前，必須用 abs() 取絕對值「還原」它！
            int now = abs(nums[i]);
            
            // 🚨 坑 2：負負得正 (記號被擦掉) 預防
            // 如果陣列裡有重複的數字 (例如兩個 2)，會重複走到同一個抽屜。
            // 所以必須先檢查，如果抽屜裡的數字還是大於 0 (正數)，我們才打上負號。
            if(nums[now-1] > 0 ){
                nums[now-1] = -nums[now-1];
            }
        }
        
        // ==========================================
        // 階段 2：抓出消失的犯人
        // ==========================================
        // 重新掃描一次陣列。
        // 如果某個抽屜的數字還是「正數」，代表從頭到尾都沒有人指過這個位置。
        for(int i = 0; i < nums.size(); i++){
            if(nums[i] > 0 ){
                // 記得把 Index 轉換回原本的數字 (i 號抽屜代表數字 i + 1)
                result.push_back(i+1);
            }
        }
        
        return result;
    }
};
