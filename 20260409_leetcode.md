class Solution {
public:
    bool isSubsequence(string s, string t) {
        
        // ==========================================
        // 第一階段：預處理 (Preprocessing) - 建立索引字典
        // ==========================================
        
        // 宣告 26 個「抽屜」，用來記錄 a~z 每個字母在 t 裡面出現的「所有真實位置 (里程數)」
        // 註：因為是照順序掃描，每個抽屜裡面的數字天生就是由小到大排好的
        vector<int> pos[26];
        
        // 掃描一次字串 t，建檔存入抽屜。時間複雜度 O(|t|)
        for(int i = 0; i < t.size(); i++){
            // t[i] - 'a' 是用來算出 0~25 的「抽屜編號」
            // i 則是這個字母在 t 裡面的「真實位置」，把它丟進抽屜裡
            pos[t[i] - 'a'].push_back(i);
        }
        
        // ==========================================
        // 第二階段：海量查詢 (Query) - 二分搜尋驗證
        // ==========================================
        
        // 記錄目前在字串 t 裡面的「真實里程碑」，確保比對順序一直是往前走
        // 初始設為 -1，代表還在起點之前
        int current_pos = -1;

        // 開始依序驗證字串 s 的每個字母。時間複雜度 O(|s| * log|t|)
        for(int i = 0; i < s.size(); i++){
            
            // 算出目前要找的字母是幾號抽屜 (0~25)
            int char_idx = s[i] - 'a';

            // it 是一個疊代器 (指標)。
            // 使用二分搜尋法 (upper_bound) 去專屬的抽屜裡，尋找「第一個大於 current_pos」的數字
            auto it = upper_bound(pos[char_idx].begin(), pos[char_idx].end(), current_pos);
            
            // 嚴謹防禦：如果疊代器指到抽屜的結尾 (end)
            // 代表在這個抽屜裡，已經找不到比目前里程數還要大的位置了，配對斷鏈！
            if(it == pos[char_idx].end()){
                return false;
            }
            else{
                // 配對成功！
                // 使用 *it (解參考) 拿出疊代器指著的「真實里程數」
                // 更新 current_pos，確保下一輪找新字母時，會從這個位置繼續往後找
				// 要注意這點
                current_pos = *it;
            }
        }
        
        // 字串 s 全部驗證完畢都沒斷鏈，代表真的是子字串
        return true;
    }
};
