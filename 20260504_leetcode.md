78. Subsets


class Solution {
public:
    void backtrack (vector<int>& nums, int index, vector<int>& subset, vector<vector<int>>& ans){
        // 1. 收集答案：
        // 把目前的籃子狀態，拍照存檔，丟進總解答陣列裡。
        ans.push_back(subset);
        
        // 2. 展開平行宇宙 (從目前的 index 開始往後選)
        for (int i = index; i < nums.size(); i++) {
            
            // 動作 A：做選擇 (選它！)
            subset.push_back(nums[i]);
            
            // 動作 B：進入下一層平行宇宙 (遞迴往下走)
            backtrack(nums, i + 1, subset, ans);
            
            // 動作 C：回溯！狀態還原 (把剛剛加進去的拿出來，退回上一步)
            subset.pop_back();
        }

    }
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<vector<int>> ans; // 準備好裝所有答案的大箱子
        vector<int> subset;      // 準備好用來嘗試組合的空籃子
        
        // 呼叫遞迴，從第 0 個數字開始做決策
        backtrack(nums, 0, subset, ans);
        
        return ans;
    }
};
