28. Find the Index of the First Occurrence in a String

class Solution {
public:
    int strStr(string haystack, string needle) {
        // 1. 轉型與儲存長度：將 size_t (無號整數) 轉為 int (有號整數)，徹底防堵 Underflow (向下溢位) 陷阱
        int size1 = haystack.size();
        int size2 = needle.size();
        
        // 2. 邊界防呆機制：如果「針」的長度大於「草堆」，實體上絕對不可能存在，直接提早結束 (Early Return)
        if(size2 > size1) return -1;
        
        // 3. 核心滑動視窗：迴圈極限精準設定為 (size1 - size2)，確保後續 substr 絕對不會發生 Out of Bounds (記憶體越界)
        for(int i = 0; i <= (size1 - size2); i++){
            // 從 index i 開始，向後切出與 needle 長度相同的片段進行比對
            if(haystack.substr(i, size2) == needle){
                return i; // 命中目標，立即回傳當前的起始 Index
            }
        }

        // 4. 冗餘檢查 (Code Review 點出)
        // 其實這裡只需要寫 `return -1;` 就可以了。
        // 因為當 i = 0 的時候，上面的 for 迴圈就已經執行過 haystack.substr(0, size2) == needle 的檢查了。
        // 如果程式能走到這一行，代表上面的迴圈「全部都沒中」，所以第 0 格也絕對不可能中，不需重複判斷。
        return -1; 
    }
};

7. Reverse Integer


class Solution {
public:
    int reverse(int x) {
        int result = 0;
        int units_digit = 0;
        
        // 【結構註解】這裡的 if-else 雖然能運作，但在實務上 x=0 本來就不會進 while，
        // 所以直接寫 while(x!=0) 即可，但為了保留你的邏輯架構，這裡維持原樣。
        if(x == 0){
            return 0;
        }
        else{
            // 【觀念解答 1：為什麼正負數可以共用同一個迴圈？】
            // 在 C++ 中，整數除法 (/) 是「向零截斷 (Truncate towards zero)」，
            // 且取餘數 (%) 的正負號會「跟著被除數走」。
            // 例如：-123 % 10 = -3。這完美符合我們要推入負數的需求，完全不需要把負號拔掉再掛回去！
            while(x != 0){
                
                // 1. 取出最後一位數 (如同硬體 Shift Register 擠出 LSB)
                units_digit = x % 10;
                
                // 【觀念解答 2：為什麼不能用 Stack？】
                // 如果用 Stack (後進先出 LIFO)：% 10 取出的順序是 1 -> 2 -> 3，
                // 放進 Stack 後再 Pop 出來，順序會變回 3 -> 2 -> 1，等於沒反轉。
                // 這裡我們直接用「擠出一個馬上塞入另一個」的 Shift 概念，達成 O(1) 空間複雜度。

                // 【觀念解答 3：為什麼不能用 Bit 反轉 (Reverse Bits)？】
                // 這題是「十進位 (Base-10)」的權重反轉 (123 -> 321)，
                // 若直接把二進位 01111011 做 Bit 反轉，轉回十進位數值會面目全非，且會破壞符號位 (Sign Bit)。
                
                // 2. 嚴謹的防呆邏輯 (預防 Overflow)
                // 【觀念解答 4：為什麼要用 INT_MAX / 10 來判斷？】
                // 在 C++ 中，有號整數溢位是「未定義行為 (Undefined Behavior)」。
                // 你絕對不能等 result * 10 爆掉之後才檢查。必須在「乘法發生之前」，
                // 先確認目前的 result 是否已經逼近 32-bit 的極限 (214748364)。
                
                // 正數極限防呆：超過 214748364，或等於 214748364 且準備推進來的數字大於 7 (極限尾數)
                if (result > INT_MAX / 10 || (result == INT_MAX / 10 && units_digit > 7)) return 0;
                // 負數極限防呆：低於 -214748364，或等於 -214748364 且準備推進來的數字小於 -8 (極限尾數)
                if (result < INT_MIN / 10 || (result == INT_MIN / 10 && units_digit < -8)) return 0;

                // 3. 推入新數字 (十進位左移一位並加上新位數)
                result = result * 10 + units_digit;
                
                // 4. 原數字縮小一位
                x /= 10;
            } 
        }
        return result;
    }
};
