class Solution {
public:
    string removeDuplicates(string s) {
        string result = ""; // 直接拿字串來當 Stack 圓筒用！
        
        for (char c : s) {
            // 如果圓筒不是空的，且圓筒最後一個字母跟手上的新字母一樣
            if (!result.empty() && result.back() == c) {
                result.pop_back(); // 消除最後一個字母
            } else {
                result.push_back(c); // 把新字母疊到最後面
            }
        }
        
        // 迴圈跑完，剩下的字串就是完美結果，而且順序還是正的！
        return result; 
    }
};

class Solution {
public:
    string removeDuplicates(string s) {
        // 【降維打擊：字串即堆疊】
        // 在 C++ 中，字串 (string) 的尾部操作非常高效，
        // 我們直接把它當作 Stack (圓筒) 來使用，省去最後重新組裝和反轉的時間。
        string result = ""; 
        
        // 依序掃描原字串中的每一個字母
        for (char c : s) {
            
            // 【嚴謹防禦與偷看比對】
            // 1. !result.empty() -> 嚴謹防禦：確保圓筒內有東西，才能去偷看最後一個字母。
            // 2. result.back() == c  -> 偷看比對：back() 就等於 Stack 的 top()，
            //    如果圓筒最頂端（字串最後面）的字母，剛好等於手上拿著的新字母 c。
            if (!result.empty() && result.back() == c) {
                
                // 配對成功！發生連續重複。
                // pop_back() 就等於 Stack 的 pop()，把圓筒最頂端的字母丟掉。
                // 同時，手上的新字母 c 也不放進去了，達成「雙雙消除」的效果。
                result.pop_back(); 
                
            } else {
                // 配對失敗，或是圓筒剛好是空的。
                // push_back(c) 就等於 Stack 的 push(c)，把新字母疊到圓筒的最上方（字串的最尾巴）。
                result.push_back(c); 
            }
        }
        
        // 【完美收合】
        // 迴圈結束後，圓筒內剩下的就是完成消除的字母。
        // 因為我們直接用字串當 Stack，而且是一路往尾巴塞，
        // 所以順序天生就是正確的，完全不需要再呼叫 reverse() 反轉！
        return result; 
    }
};
