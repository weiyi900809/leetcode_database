693. Binary Number with Alternating Bits

class Solution {
public:
    bool hasAlternatingBits(int n) {
        // ==========================================
        // 步驟 1：初始化與狀態記憶 (起手式)
        // ==========================================
        // 使用遮罩 (n & 1) 取得二進位最右邊的 bit (LSB)
        // 並將它存為「上一個 bit」的狀態基準
        int prev_bit = n & 1; 
        int curr_bit = 0;
        
        // 抓完第一個 bit 後，將數字向右推一格，準備進入審判迴圈
        n >>= 1; 
        
        // ==========================================
        // 步驟 2：位移審判迴圈
        // ==========================================
        // 當數字還沒被右移推成 0 之前，持續檢查
        while(n != 0) {
            // 取得當前最右邊的新 bit
            curr_bit = n & 1;
            
            // 【核心防禦】如果當前的 bit 竟然跟上一個 bit 一樣
            // 代表出現了連續的 00 或 11，直接判定為不合格！
            if(curr_bit == prev_bit) return false;
            
            // ==========================================
            // 步驟 3：狀態傳承與推進
            // ==========================================
            // 過關了！把當前的 bit 傳交給 prev_bit，作為下一輪的比對基準
            prev_bit = curr_bit;
            
            // 數字繼續往右推一格
            n >>= 1;
        }
        
        // 如果整個數字都被推光了，都沒觸發 false，代表完美交錯
        return true;
    }
};
