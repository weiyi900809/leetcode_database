101. Symmetric Tree

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 * int val;
 * TreeNode *left;
 * TreeNode *right;
 * TreeNode() : val(0), left(nullptr), right(nullptr) {}
 * TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 * TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */

class Solution {
public:
    bool isSymmetric(TreeNode* root) {
        // 1. 總裁的邊界防禦：如果整棵樹是一片空地，那「無」本身也是一種對稱
        if (root == nullptr) {
            return true;
        }
        
        // 2. 啟動調查：總裁自己無法跟自己比，所以把底下的「左副總」和「右副總」
        // 丟進我們額外打造的內部工具 (checkMirror) 進行雙人交叉比對
        bool is_mirror = checkMirror(root->left, root->right);
    
        return is_mirror;
    }
    
    // 內部專用輔助函式：負責判斷傳進來的兩個人 (p 和 q) 是否互為鏡像
    bool checkMirror(TreeNode* p, TreeNode* q) {
        // ==========================================
        // 第一階段：嚴謹的邊界防禦 (Base Case)
        // ==========================================
        // 情境 A：兩個人都走到盡頭 (空地)，代表這部分的虛無結構是對稱的
        if (p == nullptr && q == nullptr) {
            return true;
        }
        
        // 情境 B：程式走到這，代表兩人絕不會同時為空。
        // 此時若其中一人是空地 (另一個必定有房子)，代表結構不對稱，直接淘汰
        if (p == nullptr || q == nullptr) {
            return false;
        }
        
        // ==========================================
        // 第二階段：收集情報 (遞迴交叉調查)
        // ==========================================
        // 照鏡子的核心邏輯：必須「交叉配對」往下查
        // 1. 外側對齊外側：拿 p 的「左」部門，去跟 q 的「右」部門比對
        bool left_child_is_mirror = checkMirror(p->left, q->right);
        
        // 2. 內側對齊內側：拿 p 的「右」部門，去跟 q 的「左」部門比對
        bool right_child_is_mirror = checkMirror(p->right, q->left);

        // ==========================================
        // 第三階段：統整並回報 (Return Result)
        // ==========================================
        // 要成為完美的鏡像，必須同時滿足三個嚴苛條件：
        // 1. 這兩個長官本身的業績 (val) 一模一樣
        // 2. 且外側的調查結果為 true
        // 3. 且內側的調查結果為 true
        return (p->val == q->val) && (left_child_is_mirror) && (right_child_is_mirror);
    }
};
