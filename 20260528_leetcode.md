237. Delete Node in a Linked List

class Solution {
public:
    void deleteNode(ListNode* node) {
        /*
        這題主要是要想到說刪掉node 也可以往後面去處理進而刪掉node 我們之前學的都是往前設一個prev 讓prev的next 指向要刪掉node的next 
        */
        // 先把下一個節點的記憶體位置記下來
        ListNode* next_node = node->next;
        
        // 1. 複製靈魂：把下一個節點的值拷貝過來
        node->val = next_node->val;
        
        // 2. 重新連線：直接指向下下個節點，跳過 next_node
        node->next = next_node->next;
        
        // 3. 實體火化：手動釋放被孤立的節點記憶體
        delete next_node; 
    }
};
