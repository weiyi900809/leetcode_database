290. Word Pattern

class Solution {
public:
    bool wordPattern(string pattern, string s) {

        // ==========================================
        // 🚨 觀念 1：為什麼需要「兩本」字典？
        // ==========================================
        // 為了確保「一對一絕對映射 (Bijection)」。
        // 如果只有一本字典，會無法防禦 "abba" 對應 "dog dog dog dog" 的情況
        // (a和b都單方面綁定了dog，導致dog被劈腿)。所以必須雙向查驗！
        unordered_map<char, string> char_to_word;
        unordered_map<string, char> word_to_char;

        // ==========================================
        // 🛠️ 觀念 2：C++ 字串切割神兵利器 (String Parsing)
        // ==========================================
        // C++ 沒有內建的 split()，必須依賴 stringstream 將字串轉換為資料流。
        stringstream ss(s);
        string word;
        vector<string> words;

        // 當水龍頭 (ss) 還能根據「空白字元」流出單字時，依序存入陣列中
        while (ss >> word) {
            words.push_back(word);
        }
        
        // ==========================================
        // 🚨 觀念 3：極端防禦機制 (Early Exit)
        // ==========================================
        // 如果字母的數量跟單字的數量根本對不上 (例如 pattern="a", s="dog cat")
        // 連進入迴圈比對的資格都沒有，直接判定失敗，節省運算資源！
        if(pattern.size() != words.size()) return false;

        // 核心邏輯：One Pass 雙向海關查驗
        for(int i = 0; i < pattern.size(); i++){
            
            char c = pattern[i];
            string w = words[i];

            // ==========================================
            // 🚨 觀念 4：Hash Map 的查詢地雷 (.count vs [])
            // ==========================================
            // 絕對不能直接寫 if(char_to_word[c] == "")！
            // 在 C++ 中，直接用 [] 查詢不存在的 key，會引發「自動建檔」並塞入空字串，
            // 產生嚴重的記憶體垃圾與邏輯錯誤。必須先用 .count() 嚴謹確認存在與否！

            // --- 檢查 char -> word 的綁定情況 ---
            if (char_to_word.count(c) == 0) {
                // 狀況 A：字典裡沒這個字母 (count == 0)，代表第一次見面，大膽建檔綁定
                char_to_word[c] = w; 
            } 
            else if (char_to_word[c] != w) {
                // 狀況 B：字典裡有紀錄，且承諾對象跟眼前的單字 w 不一樣，抓包劈腿！
                return false; 
            }

            // --- 檢查 word -> char 的綁定情況 ---
            if (word_to_char.count(w) == 0) {
                // 狀況 A：字典裡沒這個單字 (count == 0)，代表第一次見面，大膽建檔綁定
                word_to_char[w] = c; 
            } 
            else if (word_to_char[w] != c) {
                // 狀況 B：字典裡有紀錄，且承諾對象跟眼前的字母 c 不一樣，抓包劈腿！
                return false; 
            }
        }
        
        // 全部查驗通過，確認是一對一完美對應！
        return true;
    }
};
