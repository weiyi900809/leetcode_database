9. Palindrome Number

class Solution {
public:
    bool isPalindrome(int x) {
        // 🛡️ 保鑣 1：負數絕對不是迴文
        if (x < 0) return false;
        
        // 🛡️ 保鑣 2：尾數是 0 的數字 (除了 0 本身) 絕對不是迴文
        if (x % 10 == 0 && x != 0) return false;

        int revertedNumber = 0;
        
        // 🎯 核心迴圈：當 x > revertedNumber 時，代表還沒走到一半
        //例如 12321是奇數長度，最後 x 會是 12，revertedNumber 會是 123 while就停止
        //例如 1221是偶數數長度，最後 x 會是 12，revertedNumber 會是 12 while就停止
        while (x > revertedNumber) {
            revertedNumber = revertedNumber * 10 + x%10 ;
            x/=10;
        }

        // ==========================================
        // 🏆 最終對決：檢查是否為迴文
        // ==========================================
        // 當迴圈停下來時，數字已經被切成兩半了。
        // 情況 A (偶數長度)：例如 1221，最後 x 會是 12，revertedNumber 也會是 12
        // 情況 B (奇數長度)：例如 12321，最後 x 會是 12，revertedNumber 會是 123
        //        (這時候只要把 revertedNumber / 10，把中間那個無關緊要的 3 砍掉就好)
        
        return x == revertedNumber || x == revertedNumber / 10;
    }
};
