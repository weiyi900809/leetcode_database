35. Search Insert Position

class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int left = 0;                    // 左邊界，從陣列開頭開始
        int right = nums.size() - 1;     // 右邊界，從陣列最後一個元素開始
        int mid = 0;                     // 用來記錄中間位置

        // 只要左邊界還沒超過右邊界，就持續做二分搜尋
        while (left <= right) {
            // 計算中間位置
            // 寫成 left + (right - left) / 2 可以避免 (left + right) 直接相加造成溢位
            mid = left + ((right - left) / 2);

            // 如果 target 剛好等於中間值，直接回傳該索引
            if (target == nums[mid]) {
                return mid;
            }
            // 如果 target 比中間值大，代表答案只可能在右半邊
            else if (target > nums[mid]) {
                left = mid + 1;
            }
            // 如果 target 比中間值小，代表答案只可能在左半邊
            else {
                right = mid - 1;
            }
        }

        // 迴圈結束時，left 會停在 target 應該插入的位置
        return left;
    }
};

121. Best Time to Buy and Sell Stock



class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int arr_size = prices.size();              // 取得價格陣列的長度
        int current_minimum_price = INT_MAX;      // 紀錄目前看過的最低價格，先設成很大的值
        int maximum_profit = 0;                   // 紀錄目前能得到的最大利潤，初始為 0

        // 逐一走訪每一天的股價
        for (int i = 0; i < arr_size; i++) {

            // 如果目前價格比先前記錄的最低價格還小
            // 就更新最低買入價格
            if (current_minimum_price > prices[i]) {
                current_minimum_price = prices[i];
            }

            // 計算如果今天賣出，能得到多少利潤
            // 如果這個利潤比目前最大利潤還大，就更新答案
            if ((prices[i] - current_minimum_price) > maximum_profit) {
                maximum_profit = (prices[i] - current_minimum_price);
            }
        }

        // 回傳最後算出的最大利潤
        return maximum_profit;
    }
};
