205. Isomorphic Strings

class Solution {
public:
    bool isIsomorphic(string s, string t) {
        
        // ==========================================
        // 🚨 坑點 1：陣列大小與 ASCII 陷阱
        // ==========================================
        // 絕對不能只開 26+26+10 大小的陣列！
        // 題目規格寫了包含「所有合法的 ASCII 字元」（包含空格、逗號、特殊符號）。
        // 所以必須開大小為 256 的陣列，才能完美涵蓋所有可能的字元（抽屜）。
        
        // ==========================================
        // 🚨 坑點 2：陣列初始化與「空值」的定義
        // ==========================================
        // 必須加上 = {0} 進行聚合初始化，清除記憶體垃圾值。
        // 在 ASCII 碼中，整數 0 對應的是 Null 字元 ('\0')。
        // 因為正常的字串絕對不會包含 '\0'，所以我們完美利用 0 
        // 來當作「這格抽屜還是空的、尚未綁定任何字元」的預設防呆標記。
        char s_to_t[256] = {0};
        char t_to_s[256] = {0};
        
        // ==========================================
        // 🚨 坑點 3：不要算次數，要「雙向映射 (One Pass)」
        // ==========================================
        // 算出現次數會被 s="aba", t="xxy" (次數一樣但結構不同) 這種測資擊潰。
        // 必須用一次走訪 (One Pass)，邊走邊查驗雙方有沒有「劈腿」。
        for(int i = 0; i < s.size(); i++){
            char temp1 = s[i];
            char temp2 = t[i];
            
            // --- 檢查 s 綁定 t 的情況 ---
            if(s_to_t[temp1] == 0) {
                // 抽屜是空的 (0) 代表第一次見面，建檔綁定
                s_to_t[temp1] = temp2; 
            } 
            else if(s_to_t[temp1] != temp2) {
                // 抽屜有紀錄，而且承諾的對象不是眼前的 temp2
                // 抓包劈腿！提早結束 (Early Exit)
                return false;          
            }

            // --- 檢查 t 綁定 s 的情況 (防禦多對一的情況) ---
            if(t_to_s[temp2] == 0) {
                t_to_s[temp2] = temp1; // 第一次見面，建檔綁定
            } 
            else if(t_to_s[temp2] != temp1) {
                return false;          // 抓包劈腿！
            }
        }
        
        
        return true; 
    }
};
