class Solution {
public:
    // =====================================================================
    // 核心函式：Parenthesis_DFS (深度優先搜尋 / 回溯法)
    // =====================================================================
    // 參數說明：
    // - n: 目標的括號對數上限。
    // - current_str: 目前正在組裝的字串 (使用 '&' 傳參考，避免每次遞迴都複製字串，極大提升效能)。
    // - left_count: 紀錄這條分支目前已經用了幾個左括號 '('。
    // - right_count: 紀錄這條分支目前已經用了幾個右括號 ')'。
    // - result: 用來收集所有合法答案的陣列 (使用 '&' 傳參考，大家共用同一個箱子)。
    void Parenthesis_DFS(int n, string& current_str, int left_count, int right_count, vector<string>& result){
        
        // -----------------------------------------------------------------
        // [終止條件 Base Case]
        // 當左右括號都恰好用完 n 個時，代表組裝完成，且一定是合法的。
        // 將組裝好的字串存入 result，並結束這層遞迴 (return 退回上一層)。
        // -----------------------------------------------------------------
        if(left_count == n && right_count == n){
            result.push_back(current_str);
            return;
        }

        // -----------------------------------------------------------------
        // [決策分支 1：嘗試放入左括號 '(']
        // 剪枝條件 (黃金法則一)：只要左括號的額度還沒用完 (< n)，就隨時可以放。
        // -----------------------------------------------------------------
        if (left_count < n) {
            // 步驟一 (做選擇)：將 '(' 塞入字串尾端
            current_str.push_back('(');
            
            // 步驟二 (進入遞迴)：帶著更新後的狀態 (left_count + 1) 繼續往下探勘
            Parenthesis_DFS(n, current_str, left_count + 1, right_count, result); 
            
            // 步驟三 (復原狀態 Undo)：探勘結束退回來時，必須把剛才塞進去的 '(' 拔掉，
            // 讓 current_str 恢復原狀，才能乾淨地去走下一個分支。
            current_str.pop_back();
        }
        
        // -----------------------------------------------------------------
        // [決策分支 2：嘗試放入右括號 ')']
        // 剪枝條件 (黃金法則二)：目前的右括號數量必須「嚴格小於」左括號數量，
        // 否則放上去就會變成不合法的字串 (例如 "())")。
        // -----------------------------------------------------------------
        if (right_count < left_count) {
            // 步驟一 (做選擇)：將 ')' 塞入字串尾端
            current_str.push_back(')');
            
            // 步驟二 (進入遞迴)：帶著更新後的狀態 (right_count + 1) 繼續往下探勘
            Parenthesis_DFS(n, current_str, left_count, right_count + 1, result); 
            
            // 步驟三 (復原狀態 Undo)：探勘結束退回來時，把剛才塞進去的 ')' 拔掉。
            current_str.pop_back();
        }
    }

    // =====================================================================
    // 主函式：generateParenthesis
    // =====================================================================
    vector<string> generateParenthesis(int n) {
        // [架構師寫法：區域變數防護]
        // 將 result 宣告在函式內部，確保每一次執行 generateParenthesis 時，
        // 都是一個全新的、乾淨的 vector。這能避免 LeetCode 重複使用 Solution 物件
        // 所造成的髒資料殘留 (全域變數陷阱)。
        vector<string> result; 
        
        // 準備一個初始的空字串，準備進行遞迴組裝
        string current_str = "";
        
        // 呼叫 DFS 開始探勘，初始的 left_count 和 right_count 都是 0
        Parenthesis_DFS(n, current_str, 0, 0, result);
        
        // DFS 執行完畢後，result 裡面已經裝滿了所有合法的括號組合，直接回傳。
        return result;
    }
};
