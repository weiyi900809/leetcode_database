48. Rotate Image


class Solution {
public:
    void rotate(vector<vector<int>>& matrix) {
        
        int n = matrix[0].size();

        // ==========================================
        // 步驟 1：對角線翻轉 (Transpose)
        // ==========================================
        // 將矩陣沿著「左上到右下」的主對角線進行對稱交換
        for (int i = 0; i < n; i++) {
            // 【關鍵】j 必須從 i 開始，確保只掃描右上半部的三角形區域，
            // 避免已經交換過的元素在後續迴圈又被換回原位。
            for (int j = i; j < n; j++) {
                swap(matrix[i][j], matrix[j][i]);
            }
        }
        
        // ==========================================
        // 步驟 2：水平翻轉 (Reverse)
        // ==========================================
        // 將每一列 (Row) 的元素以垂直中線為基準，左右對稱交換
        int head = 0;
        int tail = n - 1;
        
        // 使用雙指標，從左右兩側向中間逼近
        while (head < tail) {
            // 針對每一列 (Row i) 執行左右交換
            for (int i = 0; i < n; i++) {
                // matrix[i][head] 代表左側元素
                // matrix[i][tail] 代表右側元素
                swap(matrix[i][head], matrix[i][tail]);
            }
            head++;
            tail--;
        }
    }
};
