208. Implement Trie (Prefix Tree)



struct TrieNode {
    // 陣列：準備 26 個指標，對應 a~z 的 26 條分岔路
    // 如果 children[0] 不是 nullptr，代表有 'a' 這條路可以走
    TrieNode* children[26]; 
    
    // 布林值：用來記錄走到這裡算不算是一個完整的單字
    bool isEndOfWord;

    // 建構子 (初始化)
    TrieNode() {
        isEndOfWord = false;
        for (int i = 0; i < 26; i++) {
            children[i] = nullptr; // 一開始這 26 條分岔路都不存在
        }
    }
};
/*
在 C++ 裡，當你宣告 struct TrieNode { ... }; 的那一瞬間，編譯器就會自動把 TrieNode 註冊成一個正式的資料型態！
你完全不需要加 typedef，也不需要每次都打 struct TrieNode。
直接俐落地寫下 [TrieNode* curr;] , C++ 編譯器完全看得懂。

*/
class Trie {
private:
    // 宣告一個指標，作為整棵字典樹的起點
    TrieNode* root; 

public:
    Trie() {
        // 當 Trie 被建立時，立刻產生一個空的起點節點
        root = new TrieNode(); 
    }
    
    void insert(string word) {
        // 1. 派出替身指標 curr，一開始讓它站在起點 root
        TrieNode* curr = root; 

        // 2. 用你說的 for loop 去跑 string 的每一個 character
        for (int i = 0; i < word.length(); i++) {
            char ch = word[i];
            
            // 將字元轉換成 0~25 的陣列索引 (例如 'a'->0, 'b'->1)
            int index = ch - 'a'; 

            // 3. 檢查「存進去」的邏輯：
            // 如果目前這個節點 curr 底下，通往 index 的那條路是空的 (nullptr)
            // 檢查通往 index (例如 'c' 算出來的 2) 的那個洞是不是空的
            if (curr->children[index] == nullptr) { 
                curr->children[index] = new TrieNode(); 
            }

            // 4. 不管路是原本就有的，還是剛剛新建的，我們都要讓替身 curr 「往下走一步」
            curr = curr->children[index];
        }

        // 5. 存到最後一個 character 了，迴圈結束！
        curr->isEndOfWord=true;
    }
    
    bool search(string word) {
        // 派出替身指標 curr，一開始讓它站在起點 root
        TrieNode* curr = root; 

        // 用你說的 for loop 去跑 string 的每一個 character
        for (int i = 0; i < word.length(); i++) {
            char ch = word[i];
            
            // 將字元轉換成 0~25 的陣列索引 (例如 'a'->0, 'b'->1)
            int index = ch - 'a'; 

            // 如果發現對應的字母路徑不存在 (路斷了)，代表字典裡絕對沒有這個單字
            if (curr->children[index] == nullptr) { 
                return false;
            }

            // 路徑存在，將指標往下移動到該字母的節點
            curr = curr->children[index];
        }
        // 迴圈平安結束，代表字串的路徑都走完了。
        // 檢查最後停留的這個節點，是否被標記為一個「完整單字的結尾」
        return curr->isEndOfWord;

    }
    
    bool startsWith(string prefix) {
        // 派出替身指標 curr，一開始讓它站在起點 root
        TrieNode* curr = root; 

        // 用你說的 for loop 去跑 string 的每一個 character
        for (int i = 0; i < prefix.length(); i++) {
            char ch = prefix[i];
            
            // 將字元轉換成 0~25 的陣列索引 (例如 'a'->0, 'b'->1)
            int index = ch - 'a'; 

            // 如果發現對應的字母路徑不存在 (路斷了)，代表字典裡絕對沒有這個prefix
            if (curr->children[index] == nullptr) { 
                return false;
            }

            // 路徑存在，將指標往下移動到該字母的節點
            curr = curr->children[index];
        }
        // 迴圈平安結束！這代表 prefix 的每一個字母都在樹上找到了對應的路徑。
        // 因為只是找「前綴」，所以不管這個節點有沒有被插旗子 (isEndOfWord)，都直接回傳 true
        return true;

    }
};
