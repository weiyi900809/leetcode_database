1046. Last Stone Weight


class Solution {
public:
    int lastStoneWeight(vector<int>& stones) {
        
        // ==========================================
        // 階段 1：建立魔法袋 (Max Heap)
        // ==========================================
        // priority_queue 預設就是由大到小排列 (Max Heap)。
        // 任何丟進去的數字，它都會在 O(log N) 的時間內自動排好，確保 top() 永遠是最大值。
        priority_queue<int> pq;
        
        // 將所有石頭倒進魔法袋中
        for(int i = 0; i < stones.size(); i++){
            pq.push(stones[i]);
        }
        
        // ==========================================
        // 階段 2：互砸擂台賽
        // ==========================================
        // 🚨 坑點 1：為什麼條件是 pq.size() > 1？
        // 因為每次互砸都需要「兩顆」石頭。如果只剩 1 顆或 0 顆，遊戲就該結束了。
        while( pq.size() > 1 ){
            
            // 拿出全場最重的石頭 (a)，然後從袋子裡移除
            int a = pq.top();
            pq.pop();
            
            // 拿出全場第二重的石頭 (b)，然後從袋子裡移除
            int b = pq.top();
            pq.pop();
            
            // 🚨 坑點 2：Stack 與 Heap 的核心差異
            // 如果當初用 Stack + Sort，這裡產生新石頭塞回去後，陣列的「單調性」就被破壞了。
            // 但使用 priority_queue，我們只要把剩下的重量 (a - b) 丟回去，它會自動安插到對的位置。
            
            // 🚨 坑點 3：等於的情況
            // 只有當 a > b 時才會有剩下的石頭。如果 a == b，兩顆都會粉碎，什麼都不用 push。
            // 你的寫法精準地避開了 push(0) 進去佔空間的無謂操作！
            if( a > b ){
                pq.push(a - b);
            }
        }
        
        // ==========================================
        // 階段 3：結算輸出
        // ==========================================
        // 🚨 坑點 4：全數粉碎的極端測資 (Edge Case)
        // 如果最後兩顆石頭重量剛好一樣，互砸後袋子會變成「完全空的」 (pq.size() == 0)。
        // 這時如果直接呼叫 pq.top() 程式會當場崩潰！
        // 你用三元運算子完美擋下了這個致命傷：如果有剩就回傳 top()，沒剩就回傳 0。
        return pq.size() > 0 ? pq.top() : 0;
    }
};
