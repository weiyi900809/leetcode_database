876. Middle of the Linked List

/**
 * Definition for singly-linked list.
 * struct ListNode {
 * int val;
 * ListNode *next;
 * ListNode() : val(0), next(nullptr) {}
 * ListNode(int x) : val(x), next(nullptr) {}
 * ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        // 初始化快慢指標，兩者一開始都站在起跑點 (Head)
        ListNode* fast = head;
        ListNode* slow = head;

        // 【核心邊界防禦】
        // 條件 1: fast != NULL 確保在「偶數」長度時，兔子掉出懸崖後能安全停下。
        // 條件 2: fast->next != NULL 確保在「奇數」長度時，兔子踩在最後一個節點上能安全停下。
        while(fast != NULL && fast->next != NULL){
            // 兔子每次跳兩步 (必須有上面的防護罩，這裡的 ->next->next 才不會當機)
            fast = fast->next->next;
            
            // 烏龜每次走一步
            slow = slow->next;
        }
        
        // 迴圈結束時，兔子到底了，烏龜必定精準停在正中間
        return slow;
    }
};
