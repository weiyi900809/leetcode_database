383. Ransom Note


class Solution {
public:
    bool canConstruct(string ransomNote, string magazine) {
        
        // ==========================================
        // 🚨 觀念 1：精準的陣列大小與初始化
        // ==========================================
        // 題目限制只有「小寫英文字母」，所以精準開 26 格抽屜即可，達成 O(1) 空間。
        // 加上 = {0} 完美清除記憶體垃圾值，把所有字母的初始庫存歸零。
        int maga_counter[26] = {0};
        
        // ==========================================
        // 📦 階段 1：進貨 (統計 magazine 的可用字母)
        // ==========================================
        // 走訪一次雜誌字串。時間複雜度 O(M)
        for(int i = 0; i < magazine.size(); i++){
            // 利用 magazine[i] - 'a' 把字元 'a'~'z' 轉換成 0~25 的抽屜編號。
            // 找到對應抽屜後，將庫存數量 +1 (進貨)
            maga_counter[magazine[i] - 'a']++;
        }
        
        // ==========================================
        // 🪓 階段 2：扣庫存 (驗證 ransomNote 是否能被拼湊)
        // ==========================================
        // 走訪一次勒索信字串。時間複雜度 O(N)
        for(int i = 0; i < ransomNote.size(); i++){
            
            // 查閱目前需要的這個字母，它的庫存剩多少？
            // 🚨 觀念 2：提早攔截 (Early Exit)
            // 如果發現這個字母的庫存已經是 0 了 (代表沒貨了，或者根本沒進貨過)
            if( maga_counter[ransomNote[i] - 'a'] == 0 ){
                return false; // 庫存不足，無法拼出勒索信，直接抓包！
            }
            else{
                // 如果還有庫存，就把庫存數量 -1 (消耗掉一個字母)
                maga_counter[ransomNote[i] - 'a']--;
            }
        }
        
        // ==========================================
        // 🏆 階段 3：全部驗證通過
        // ==========================================
        // 勒索信的每一個字母都成功從倉庫裡扣除庫存，代表字母完全夠用！
        return true;
    }
};
