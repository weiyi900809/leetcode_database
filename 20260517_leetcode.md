415. Add Strings



class Solution {
public:
    string addStrings(string num1, string num2) {
        string ans;
        int carry = 0;
        
        // 從字串的最尾巴（個位數）開始往前掃描
        int i = num1.size() - 1;
        int j = num2.size() - 1;
        
        // ==========================================
        // 🚨 坑點 1：遺落的最高位進位 (The Lost Carry)
        // ==========================================
        // 迴圈條件必須是「三位一體」。
        // 除了判斷 i 和 j 是否走完，絕對不能漏掉 carry > 0。
        // 否則在計算 99 + 1 時，最後的進位 1 會被吃掉，答案變成 00。
        while (i >= 0 || j >= 0 || carry > 0) {
            
            // ==========================================
            // 🚨 坑點 2：陣列越界與動態補零 (Virtual Zero)
            // ==========================================
            // 字串長度可能不同（例如 999 + 1）。
            // 絕對不能無腦讀取 num1[i]，必須先用 i >= 0 檢查。
            // 如果越界了，就用三元運算子補上 0，讓直式加法能完美對齊。
            // 同時利用 c - '0' 將字元極速轉為整數。
            int n1 = (i >= 0) ? num1[i] - '0' : 0;
            int n2 = (j >= 0) ? num2[j] - '0' : 0;
            
            // 將兩個數字與「上一回合傳下來的進位」相加
            int sum = n1 + n2 + carry;
            
            // 計算要丟給下一回合的進位 (例如 sum = 14，carry = 1)
            carry = sum / 10;
            
            
            ans += (sum % 10) + '0' ; 
            
            // 指標往前推進 (往高位數走)
            i--;
            j--;
        }
        
        // ==========================================
        // 🚨 坑點 3：忘記反轉字串
        // ==========================================
        // 因為我們是從個位數開始計算並加到 ans 字串的尾巴，
        // 導致越小的位數排在字串越前面 (例如 12 + 9 會算出 "12")。
        // 最後必須呼叫 reverse 將整個字串翻轉回來變成正確的 "21"。
        reverse(ans.begin(), ans.end());
        
        return ans;
    }
};
