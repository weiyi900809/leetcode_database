459. Repeated Substring Pattern








方法一：


class Solution {
public:
    bool repeatedSubstringPattern(string s) {
        int n = s.size();
        
        // ==========================================
        // 步驟 1：界定搜尋範圍 (外層迴圈)
        // ==========================================
        // 變數 i 代表我們「猜測的子字串長度」。
        // 長度最短是 1，最長絕對不可能超過原字串的一半 (n / 2)。
        // 這樣直接省下了一半的無效窮舉時間。
        for(int i = 1; i <= n / 2; i++){
            
            // ==========================================
            // 步驟 2：第一道防線 (數學倍數過濾)
            // ==========================================
            // 這是最重要的效能優化！
            // 如果原字串長度無法被 i 整除，代表這個長度根本不可能填滿原字串。
            // 只有當 n % i == 0 時，我們才進行昂貴的字串操作。
            if(n % i == 0){
                
                // ==========================================
                // 步驟 3：字串切片與組裝 (核心驗證)
                // ==========================================
                // 將最前面的 i 個字元切下來，當作「重複的積木」。
                string temp = s.substr(0, i);
                
                // 計算這個積木需要被複製幾次
                int times = n / i;
                string test_str = "";
                
                // 內層迴圈：進行積木的物理拼湊
                for(int j = 0; j < times; j++){
                    test_str += temp;
                }
                
                // ==========================================
                // 步驟 4：比對與收網
                // ==========================================
                // 檢查拼湊出來的測試字串，是否跟原字串完全一模一樣。
                // 如果是，立刻鳴金收兵。
                if(test_str == s) return true;
            }
        }
        
        // 如果外層迴圈把所有可能的長度都試過了，還是找不到，代表它不是重複子字串。
        return false;
    }
};

class Solution {
public:
    bool repeatedSubstringPattern(string s) {
        int n = s.size();
        
        // ==========================================
        // 步驟 1：倍增字串 (拓樸展開)
        // ==========================================
        // 將原字串拼接一次，創造出包含所有可能「旋轉狀態」的新字串。
        // 例如：s = "abcabc" (長度 6) -> test_str = "abcabcabcabc" (長度 12)
        string test_str = s + s;
        
        // ==========================================
        // 步驟 2：破壞頭尾防線 (去頭去尾)
        // ==========================================
        // 從 Index 1 開始切，拿掉最前面和最後面各 1 個字元。
        // 計算要保留的長度：總長度 (2n) - 砍掉頭 (1) - 砍掉尾 (1) = 2n - 2。
        // 如果 s 是由重複子字串組成的，那麼中間殘留的區塊一定還夾著一個完整的 s。
        test_str = test_str.substr(1, (2 * n) - 2); 
        
        // ==========================================
        // 步驟 3：在殘骸中尋找 (核心驗證)
        // ==========================================
        // 使用 C++ 內建的 find 尋找 s。
        // 必須嚴格比對回傳值是否「不等於 string::npos」，來確認是否真的有找到。
        return test_str.find(s) != string::npos;
    }
};
