438. Find All Anagrams in a String


class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        vector<int> ans;
        
        // ==========================================
        // 🚨 坑點 1：字串比較的致命陷阱 (字典序 vs 長度)
        // ==========================================
        // 之前誤寫成 if(s < p)，這在 C++ 會觸發「字典序」比較，而非長度比較。
        // 例如 "z" > "abc" 會成立，導致防呆機制被繞過，後續引發陣列越界死當。
        // ✅ 嚴謹的寫法：明確呼叫 .size() 來比較真實長度。
        if(s.size() < p.size()) return ans;
        
        // 1. 製作通緝令 (長度 26 的 Hash 陣列)
        vector<int> target_hash(26, 0);
        int p_length = p.size();
        int s_length = s.size();
        
        for(char c : p){
            // 透過減去 ASCII 'a'，將小寫字母完美映射到 0~25 的 index
            target_hash[c - 'a']++;
        }
        
        // 2. 建立初始放大鏡 (對準 s 的前 p_length 個字元)
        vector<int> window_hash(26, 0);
        for(int i = 0; i < p_length; i++){
            char c = s[i];
            window_hash[c - 'a']++;
        }
        
        // 3. 第一次盤查 (起始位置 index 0)
        // C++ vector 內建 == 運算子，直接在底層做極速記憶體比對
        if (target_hash == window_hash) ans.push_back(0);
        
        // ==========================================
        // 🚨 坑點 2：滑動視窗的邊界算術地雷 (Off-by-one Error)
        // ==========================================
        // 之前嘗試用長度去推算迴圈終點 (tail <= (s_length - p_length) + 1)，
        // 這種加減算術極度容易算錯，導致最後一個字母漏抓。
        // ✅ 嚴謹的寫法：因為 tail 代表「即將吃進來的新字元」，
        // 所以只要 tail 還沒超出字串總長 (tail < s_length)，就繼續吃，無腦且絕對安全！
        for(int head = 1, tail = p_length; tail < s_length; tail++, head++){
            
            // 動作 A：吐出舊字母 (放大鏡左邊滑出去的那個字，index 為 head - 1)
            char c = s[head - 1];
            window_hash[c - 'a']--;
            
            // 動作 B：吃進新字母 (放大鏡右邊剛罩住的那個字，index 為 tail)
            c = s[tail];
            window_hash[c - 'a']++;
            
            // 動作 C：再次盤查，如果吻合，記錄目前的起點 (head)
            if (target_hash == window_hash) ans.push_back(head);
        }
        
        return ans;
    }
};
