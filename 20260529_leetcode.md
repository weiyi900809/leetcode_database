203. Remove Linked List Elements


/**
 * Definition for singly-linked list.
 * struct ListNode {
 * int val;
 * ListNode *next;
 * ListNode() : val(0), next(nullptr) {}
 * ListNode(int x) : val(x), next(nullptr) {}
 * ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* removeElements(ListNode* head, int val) {
        // 【核心技巧：虛擬頭節點 Dummy Node】
        // 將 dummy 宣告在 Stack (堆疊) 上而非使用 new，函式結束時會自動銷毀，絕對不會 Memory Leak。
        // 它的作用是成為所有節點（包含原 Head）永遠的「前一個節點」，避免處理 Head 被刪除的特例。
        ListNode dummy(0); 
        dummy.next = head; 
        
        // 雙指標起跑點：prev 站在 Dummy，curr 站在真正的起點 Head
        ListNode* prev = &dummy; 
        ListNode* curr = prev->next; 
        
        // 只要還沒走到串列尾端，就繼續掃描
        while(curr != NULL){
            
            // 情況 A：抓到了！這個節點必須被刪除
            if(curr->val == val){
                // 1. 物理繞道：把 prev 的箭頭直接連到 curr 的下一格
                prev->next = curr->next;
                
                // 2. 準備火化：先把準備要刪除的 curr 記在 temp 裡
                ListNode* temp = curr;
                
                // 3. 探路者前進：curr 走到下一格繼續任務
                // 【關鍵防護】注意！這裡 prev 絕對不能動！它必須留在原地，監控新接上來的節點是不是也要刪除。
                curr = curr->next;
                
                // 4. 釋放記憶體：徹底銷毀剛剛拔下來的孤兒節點
                delete temp;
            }
            // 情況 B：安全！這個節點不需要刪除
            else{
                // 因為確定安全，所以 prev 可以安心地跟著 curr 一起往前走一步
                prev = curr;
                curr = curr->next;
            }
        }
        
        // 回傳 Dummy 後面接的節點，這就是清洗完畢後的全新 Head
        return dummy.next;
    }
};
