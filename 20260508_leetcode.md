338. Counting Bits

class Solution {
public:
    vector<int> countBits(int n) {
        
        // ==========================================
        // 🚨 坑點：動態陣列的「雙重初始化」地雷
        // ==========================================
        // 當你寫下 vector<int> ans(n + 1) 時，C++ 已經做完兩件事：
        // 1. 開闢了 n + 1 個格子。
        // 2. 把所有格子都填滿了預設值 0。
        // 
        // ❌ 之前的錯誤寫法：ans.push_back(0);
        // push_back 的物理意義是「在陣列的最尾巴『新增』一個格子」。
        // 如果你已經宣告了長度為 n+1，又 push_back，總長度會變成 n+2！
        // 這樣回傳給 LeetCode 檢查時，就會因為長度不符而直接拿到 Wrong Answer。
        // ✅ 正確做法：既然格子開好了且已經是 0，就直接進入迴圈修改它。
        vector<int> ans(n + 1);
        
        // ==========================================
        // 核心邏輯：位元 DP 狀態轉移
        // ==========================================
        // 從數字 1 開始算到 n (因為 ans[0] 已經預設為 0，不用算)
        for(int i = 1; i <= n; i++) {
            
            // 神級狀態轉移公式：
            // 1. ans[i >> 1] : 把最後一個 bit 砍掉 (也就是除以 2)，直接去查表看過去的答案。
            // 2. (i & 1)     : 檢查剛剛砍掉的那個尾巴是不是 1 (奇數為 1，偶數為 0)。
            // 兩者相加，就是現在這個數字擁有的 1 的總數！
            ans[i] = ans[i >> 1] + (i & 1);
        }
        
        return ans;
    }
};
