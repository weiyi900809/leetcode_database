844. Backspace String Compare


class Solution {
public:
    // 輔助函式：模擬鍵盤輸入並處理退匣鍵，回傳最終顯示的真實字串
    string getFinalString(string str) {
        stack<char> st; // 宣告一個字元堆疊，用來暫存打上去的字母
        
        // 走訪字串中的每一個字元
        for (char c : str) {
            if (c != '#') {
                // 1. 遇到正常英文字母：無條件放進 Stack 頂端
                st.push(c);
            } else {
                // 2. 遇到退匣鍵 '#'：
                // 【嚴謹的防禦機制】必須先確認 Stack 裡面有東西 (!st.empty()) 才能刪除。
                // 否則在空的 Stack 上執行 pop() 會導致 Segmentation Fault (記憶體區段錯誤)。
                if (!st.empty()) {
                    st.pop();
                }
            }
        }

        string result = "";
        // 3. 將 Stack 內的字母依序取出，組合成字串
        while (!st.empty()) {
            char temp;
            temp = st.top(); // 讀取最頂端的字母
            result += temp;  // 附加到結果字串的尾巴
            st.pop();        // 確實將字母從堆疊中移除
        }
        
        // 4. 【反轉陷阱修正】
        // 因為 Stack 的物理特性是後進先出 (LIFO)，倒出來的字串順序會是顛倒的 (例如 "abc" 會變成 "cba")。
        // 基於嚴謹原則，必須使用 reverse 將其反轉回原本正確的順序。
        reverse(result.begin(), result.end());
        
        return result;
    }
    
    // 主程式：比對兩個字串處理後的結果
    bool backspaceCompare(string s, string t) {
        // 分別計算兩個字串經過退匣處理後的最終真實結果
        string ans1 = getFinalString(s);
        string ans2 = getFinalString(t);
        
        // 直接比對兩者是否完全相同，相同回傳 true，反之 false
        return ans1 == ans2;
    }
};
