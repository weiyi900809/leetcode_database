13. Roman to Integer

class Solution {
public:
    int translate(char c) {
        switch (c) {
            case 'I': return 1;
            case 'V': return 5;
            case 'X': return 10;
            case 'L': return 50;
            case 'C': return 100;
            case 'D': return 500;
            case 'M': return 1000;
            default:  return 0; 
        }
    }

    int romanToInt(string s) {
        int result=0;
        for(int i=0;i<s.length();i++){
            
            if( i!=(s.length()-1) ){
                
                if( translate(s[i])  < translate(s[i+1]) ){
                    result -= translate(s[i]);
                }
                else{
                    result += translate(s[i]);
                }
            }
            else{
                result += translate(s[i]);
            }
        }
        return result;
    }
};

class Solution {
public:
    int translate(char c) {
        switch (c) {
            case 'I': return 1;
            case 'V': return 5;
            case 'X': return 10;
            case 'L': return 50;
            case 'C': return 100;
            case 'D': return 500;
            case 'M': return 1000;
            default:  return 0; 
        }
    }

    int romanToInt(string s) {
        int result = 0;
        int prev_val = 0; // 用來記錄「右邊那個數字」的大小

        // 【逆向掃描】從字串的最結尾 (尾巴) 開始往左讀
        for (int i = s.length() - 1; i >= 0; i--) {
            
            // 【消除重複運算】把當前翻譯的數字先存起來
            int current_val = translate(s[i]);
            
            // 比較器：如果當前數字比右邊的數字小 (例如 I 在 V 左邊)，觸發扣除機制
            if (current_val < prev_val) {
                result -= current_val;
            } 
            // 正常情況：當前數字大於等於右邊，或是剛好是最右邊的第一個數字，直接加總
            else {
                result += current_val;
            }
            
            // 迴圈結束前，把自己變成「右邊的數字」，供下一輪往左掃描時做比較
            prev_val = current_val;
        }
        
        return result;
    }
};
