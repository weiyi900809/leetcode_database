322. Coin Change

class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        // =====================================================================
        // 階段 1：定義狀態與初始配置 (Base Case)
        // =====================================================================
        // 【防禦性設計：安全的無限大】
        // 為什麼不用 INT_MAX？因為後續做 + 1 運算時，INT_MAX + 1 會發生整數溢位 (Overflow) 變成負數。
        // 邏輯上，就算全部用 1 元硬幣來湊，最多也只需要 amount 枚。
        // 因此，amount + 1 是一個絕對不可能達成的「安全無限大」，且計算時絕對不會溢位。
        int safe_infinity = amount + 1;
        
        // 定義 DP 陣列：dp[i] 代表「湊出金額 i 所需要的最少硬幣數量」。
        // 預設所有金額都需要「無限大」枚硬幣。
        vector<int> dp(amount + 1, safe_infinity);
        
        // 初始狀態：湊出 0 元需要 0 枚硬幣。這是整個 DP 推導的地基。
        dp[0] = 0;
        
        // =====================================================================
        // 階段 2：由底向上推進 (Bottom-Up 狀態轉移)
        // =====================================================================
        // 外層迴圈：從小到大，依序計算每個金額的最佳解 (沿著黑板一路往後算)
        for(int i = 1; i <= amount; i++){
            
            // 內層迴圈：在當前的金額 i 之下，嘗試口袋裡的每一種硬幣
            for(int j = 0; j < coins.size(); j++){
                
                // 拿出目前要測試的硬幣面額
                int coin_now = coins[j];

                // 【記憶體防護】只有當「目標金額 (i) >= 硬幣面額」時，這枚硬幣才放得下，
                // 也才能確保去查閱 dp[i - coin_now] 時，Index 不會變成負數。
                if (i >= coin_now) {
                    
                    // 【核心狀態轉移方程式】
                    // dp[i] (目前該金額的最佳解) 
                    //   vs 
                    // dp[i - coin_now] + 1 (扣掉這枚硬幣後的歷史最佳解，加上現在手裡這 1 枚)
                    // 兩者取最小值，更新紀錄。
                    dp[i] = min(dp[i], dp[i - coin_now] + 1);
                }
            }
        }
        
        // =====================================================================
        // 階段 3：最終驗收
        // =====================================================================
        // 如果迴圈跑完，目標金額的答案還是初始的「安全無限大」，
        // 代表所有硬幣組合都湊不出這個數字，依題目規定回傳 -1。
        if (dp[amount] == safe_infinity) {
            return -1;
        } 
        
        // 否則回傳算出的最少硬幣枚數
        return dp[amount];
    }
};
