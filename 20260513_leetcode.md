142. Linked List Cycle II



/**
 * Definition for singly-linked list.
 * struct ListNode {
 * int val;
 * ListNode *next;
 * ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        
        // ==========================================
        // 🏁 第一階段：龜兔賽跑 (尋找相遇點)
        // ==========================================
        // 1. 初始化快慢指標，起點都在 head
        ListNode *slow = head;
        ListNode *fast = head;
        bool is_cycle = false;
        
        // ==========================================
        // 🚨 坑點 1：兔子暴斃的防呆邊界 (Segmentation Fault)
        // ==========================================
        // 兔子每次要跨「兩步 (fast->next->next)」。
        // 如果不檢查 fast->next 是否存在，當 fast 站在最後一個節點時，
        // 呼叫 fast->next->next 就會試圖讀取 NULL 的下一個東西，直接死當！
        while (fast != NULL && fast->next != NULL) {
            
            // 烏龜走一步，兔子走兩步
            slow = slow->next;
            fast = fast->next->next;
            
            // 狀態機崩潰偵測：如果在某個 Clock 週期，兔子和烏龜在同一個記憶體位址相遇了，
            // 代表他們陷入了無窮迴圈 (Cycle)
            if (fast == slow) {
                is_cycle = true;
                break; // 抓到相遇點，立刻停止賽跑！
            }
        }
        
        // ==========================================
        // 🚨 坑點 2：直線賽道的誤判
        // ==========================================
        // 如果上面的 while 迴圈是因為 fast 跑到終點 (NULL) 而結束，
        // 代表這是一條沒有迴圈的死胡同。必須提早 return，否則進入第二階段會無窮迴圈。
        if (!is_cycle) return nullptr;
        
        // ==========================================
        // 📐 第二階段：跨界數學推導 (尋找迴圈入口)
        // ==========================================
        // 數學神級公式回顧：
        // L (起點到入口的距離) = nC - X (相遇點繼續走過迴圈然後到入口的距離)
        // 物理意義：「把一隻烏龜放回起點，另一隻留在相遇點，兩者用相同速度前進，絕對會在入口相撞！」
        
        // 拔起一隻烏龜 (slow)，啟動時光倒流，丟回整個世界的起點 (head)
        slow = head;
        
        // 另一隻烏龜 (原本的 fast) 留在剛剛的相遇點。
        // 現在開始，不准用跑的，兩隻烏龜「同步每次只走一步」。
        while (fast != slow) {
            slow = slow->next;
            fast = fast->next;
        }
        
        // 兩隻烏龜相撞的那一刻，腳下踩著的絕對就是迴圈入口！
        return slow;
    }
};
