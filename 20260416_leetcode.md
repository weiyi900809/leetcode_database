法一：
class Solution {
public:
    bool containsNearbyDuplicate(vector<int>& nums, int k) {
        unordered_map<int, int> nums_to_index;
        
        for (int i = 0; i < nums.size(); i++) {
            
            // 1. 把 find 找到的結果 (指標) 存進一個變數 it 裡 (auto 會自動推斷型別)
            auto it = nums_to_index.find(nums[i]); 
            
            // 2. 如果 it 不是指向結尾，代表找到了！
            if (it != nums_to_index.end()) { 
                
                // 3. it->second 就可以直接拿出字典裡存的 Value (也就是 index)
                // 完全不需要再用 nums_to_index[nums[i]] 重新找一次
                if (i - it->second <= k) {
                    return true;
                }
            }
            
            // 無論如何，更新當前數字的最新位置
            nums_to_index[nums[i]] = i; 
        }
        return false;
    }
};


法二：
class Solution {
public:
    bool containsNearbyDuplicate(vector<int>& nums, int k) {
        unordered_set<int> nums_in_window;
        
        for (int i = 0; i < nums.size(); i++) {
            
            // 1. 檢查抓包：這個數字在我們最近的 k 步以內出現過嗎？
            if(nums_in_window.count(nums[i]) > 0){
                return true; 
            }
            
            // 2. 入庫登記：把現在這個數字加進包廂
            nums_in_window.insert(nums[i]);
            
            // 3. 維持秩序：如果包廂人數超過 k 了，把「最舊」的那個人踢掉
            if (nums_in_window.size() > k) {
                nums_in_window.erase(nums[i - k]); 
            }
        }
        return false;
    }
};
