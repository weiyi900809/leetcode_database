242. Valid Anagram



class Solution {
public:
    bool isAnagram(string s, string t) {
        
        // ==========================================
        // 👑 核心架構：動態字典 (解決 Unicode 記憶體浪費)
        // ==========================================
        // 如果用陣列 int[150000] 來裝 Unicode，會浪費極大的記憶體空間。
        // 改用 unordered_map，只有「真正出現過的字元」才會被建檔。
        // 時間複雜度維持接近 O(N)，空間複雜度降到最優的 O(出現過的獨立字元數)。
        unordered_map<char, int> unicode_count;
        
        // 🚨 防禦機制：長度不一樣，絕對不可能同構 (Early Exit)
        if(s.size() != t.size() ) return false;
        
        // ==========================================
        // 📦 階段 1：進貨 (統計 s 的字元庫存)
        // ==========================================
        for(int i = 0; i < s.size(); i++){
            // 不需要再減去 'a' 做平移了，直接把真實的字元 (包含 Unicode) 當作 Key
            unicode_count[s[i]]++;
        }
        
        // ==========================================
        // 🪓 階段 2：扣庫存與查驗 (走訪 t)
        // ==========================================
        for(int i = 0; i < t.size(); i++){
            
            // 💡 C++ Map 底層知識點：自動建檔 (Auto-Insertion)
            // 如果 t[i] 是一個從沒見過的全新字元，C++ 會自動幫它建檔並設為 0。
            // 接著判斷 0 > 0 (為 false)，完美跳進 else 執行 return false，成功抓包！
            // 所以在這裡不用 .count() 也是絕對安全的。
            
            if(unicode_count[t[i]] > 0){
                // 庫存夠用，消耗掉一個字元
                unicode_count[t[i]]--;
            }
            else{
                // 狀況 A：字元根本沒見過 (庫存為 0)
                // 狀況 B：字元見過，但庫存已經被扣光了
                // 抓包！直接判定失敗
                return false;
            }
        }
        
        // 歷經嚴格查驗都沒被 return false，確認為 Valid Anagram！
        return true;
    }
};
