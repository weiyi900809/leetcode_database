
class Solution {
public:
    int lengthOfLastWord(string s) {
        // 1. 宣告一個字串串流物件 ss，並把整串字串 s 倒進去
        stringstream ss(s); 
        
        string word;
        
        // 2. ss >> word 的意思是：「從串流中抽出一個單字，放進 word 變數裡」
        // 當串流裡面還有單字時，它會回傳 true 讓迴圈繼續。
        // 當抽出最後一個單字後，串流空了，迴圈就會自動結束。
        while (ss >> word) {
            // 裡面什麼都不用做，我們只是利用迴圈把前面的單字全部消耗掉
        }
        
        // 3. 迴圈結束時，word 裡面裝的絕對是「最後一個單字」
        return word.length();
    }
};
