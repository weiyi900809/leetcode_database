

228. Summary Ranges
class Solution {
public:
    vector<string> summaryRanges(vector<int>& nums) {
        
        // ==========================================
        // 🚨 坑 1：字串陣列的初始化地雷
        // ==========================================
        // 絕對不能寫 vector<string> result = {0}; 
        // 因為 0 會被當成 Null Pointer (空指標)，導致字串讀取崩潰當機。
        // 在 C++ 裡，不給初始值，它就會自動變成一個乾淨的空陣列！
        vector<string> result;
        
        // ==========================================
        // 🚨 坑 2：保鑣 (Guard Clause) 的站位
        // ==========================================
        // 保鑣必須站在最門口！
        // 必須先檢查陣列是否為空，才能去拿 nums[0]。
        // 如果先寫 start = nums[0]，遇到空陣列 [] 會直接 Out of Bounds 越界當機！
        if(nums.size() == 0) {
            return result;
        }
        
        // 確認安全後，錨點釘在第一個數字上
        int start = nums[0];
        
        // 字串變數宣告 (直接寫 string temp; 就會自動初始化為空字串 "")
        string temp; 
        
        // 🏃‍♂️ 探險家 i 開始往前衝
        for(int i = 0; i < nums.size(); i++){

            // ==========================================
            // 🚨 坑 3：溢位當機與短路魔法 (Short-Circuit)
            // ==========================================
            // 如果不加 i == nums.size() - 1，探險家走到最後一格時，去讀 nums[i + 1] 會越界當機。
            // 這裡利用了 || (或) 的「由左至右、短路求值」特性：
            // 當 i 走到最後一格時，左邊條件為 true，電腦就「絕對不會」去執行右邊的 nums[i + 1]，完美防禦！
            if (i == nums.size() - 1 || nums[i + 1] != nums[i] + 1) {
                
                // 📦 進入打包出貨流程
                if(start != nums[i]){
                    // 區間內有多個數字
                    temp = to_string(start) + "->" + to_string(nums[i]);
                }
                else{
                    // 區間內只有一個數字
                    temp = to_string(start);
                }
                
                result.push_back(temp);
                
                // 📍 拔起錨點，重新出發
                // 如果探險家還沒走到盡頭，就把錨點釘到下一個全新的數字上
                if (i < nums.size() - 1) {
                    start = nums[i + 1]; 
                }
            }
        }
        return result;
    }
};
