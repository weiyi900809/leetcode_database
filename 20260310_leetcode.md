1. Two Sum

class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int, int> numMap;
        int arr_size = nums.size();
        
        for(int i = 0; i < arr_size; i++){
            int complement = target - nums[i];
            
            // find() 會回傳一個迭代器(Iterator)。
            // 如果找不到，它會剛好等於 .end()；如果不等於 .end()，代表找到了！
            if (numMap.find(complement) != numMap.end()) {
                // 找到了！直接用 [] 取出 complement 對應的 index，並與當前的 i 一起回傳
                return {numMap[complement], i};
            }
            else{
                // 如果還沒遇到有緣人，就把「自己的數值」當作 Key，「自己的位置 i」當作 Value 存起來
                numMap[nums[i]] = i;
            }
        }
        
        // 嚴謹的工程師習慣：雖然題目保證「一定有一組解」，
        // 但為了避免 C++ 編譯器警告「有控制流程可能不會回傳值」，在最後補上一個空的回傳。
        return {}; 
    }
};

26. Remove Duplicates from Sorted Array


class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int arr_size = nums.size();
        
        // 嚴謹的防呆機制：如果遇到空陣列，直接回傳長度 0
        if (arr_size == 0) return 0;

        // 第一個數字 (nums[0]) 一定保留，所以寫入指標直接從 1 開始待命
        int write_ptr = 1;
        
        // 讀取指標也從 1 開始，無腦往右掃描
        for(int read_ptr = 1; read_ptr < arr_size; read_ptr++){
            // 只要現在讀到的數字，跟「上一個數字」不一樣，代表找到全新的有效資料
            if(nums[read_ptr] != nums[read_ptr-1]){
                // 把新資料覆蓋到寫入指標的位置
                nums[write_ptr] = nums[read_ptr];
                // 寫入指標往前推一格，準備迎接下一個新數字
                write_ptr++;
            }
        }
        
        // 注意這裡：為什麼是回傳 write_ptr 而不是 write_ptr + 1？
        // 因為當我們寫入第 k 個元素後，write_ptr++ 會剛好等於 k。
        // 所以迴圈結束時，write_ptr 的數值就是不重複元素的「總長度」。
        return write_ptr;
    }
};
/*
陣列的第一個數字（Index 0）絕對是不重複的第一個有效數字。
既然第一筆資料永遠是合法的起點，我們的 write_ptr 和 read_ptr 根本不需要從 0 開始，直接從 1 開始就好！
這樣不僅完美避開了 -1 越界的問題，邏輯也變得極度乾淨
