class Solution {
public:
    void gameOfLife(vector<vector<int>>& board) {
        // 取得列數 (m: rows)
        int m = board.size();

        // 取得行數 (n: columns)
        // LeetCode 這題有保證 m, n >= 1，所以直接取 board[0] 是安全的
        int n = board[0].size();

        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                int neighbor_lives=0;
                int neighbor_dies=0;
                int current_state=board[i][j];
                for (int a = -1; a <= 1; ++a) {
                    for (int b = -1; b <= 1; ++b) {
                        int r=i+a;//鄰居的列
                        int c=j+b;//鄰居的行                 
                        // 假設你正在看 board[i][j]，且算出某個鄰居的座標是 (r, c)
                        // 必須確保 r 和 c 沒有超出 0 ~ m-1 和 0 ~ n-1 的範圍
                        if(a!=0 || b!=0){
                            if (r >= 0 && r < m && c >= 0 && c < n) {
                                
                                if(board[r][c] == 1 || board[r][c] == 2 ){//鄰居是live 或是目前是live 等下是die
                                    neighbor_lives++;

                                }
                                if(board[r][c] == 0 || board[r][c] == 3 ){//鄰居是die 或是目前是die 等下是live
                                    neighbor_dies++;
                                }

                            }
                        }
                        

                    }
                }
                if(neighbor_lives > 3 && current_state == 1){ 
                    //目前是live 但鄰居超過三個是live 這個cell next state 要die 但先改成2
                    board[i][j]=2;

                }
                if(neighbor_lives == 3 && current_state == 0){ 
                    //目前是die 但鄰居剛好三個是live 這個cell next state 要live 但先改成3
                    board[i][j]=3;
                } 
                if(neighbor_lives < 2 && current_state == 1){ 
                    //目前是die 但鄰居剛好三個是live 這個cell next state 要die 但先改成2
                    board[i][j]=2;
                }    

            }
        }
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if(board[i][j]==3){ 
                    board[i][j]=1;//改回來1
                } 
                if(board[i][j]==2){ 
                    board[i][j]=0; //改回來0
                }  
            }
        }


    }
};
