class Solution {
public:
    string addBinary(string a, string b) {
        string ans;
        // 取得兩個字串最後一個字元的索引（最低位）
        int i = a.size() - 1;
        int j = b.size() - 1;
        int carry = 0; // 進位

        // 當 a 或 b 還有字元，或是最後還有進位時，繼續執行
        while (i >= 0 || j >= 0 || carry > 0) {
            int sum = carry; // 每一位的總和先加上之前的進位

            // 如果 a 還有字元，把 a 的當前字元轉成整數並加上去
            if (i >= 0) {
                sum += a[i] - '0'; // 將字元 '0' 或 '1' 轉為整數 0 或 1
                i--;               // 往前移一位
            }
            
            // 如果 b 還有字元，把 b 的當前字元轉成整數並加上去
            if (j >= 0) {
                sum += b[j] - '0';
                j--;
            }

            // 計算當前位數要留下的值 (sum % 2)，並加到 ans 後面
            ans+= (sum % 2) + '0'; // 轉回字元後加入

            // 計算新的進位 (如果 sum 是 2 或 3，carry 就是 1)
            carry = sum / 2; 
        }

        // 因為我們是從低位加到高位（從後往前），最後要把結果反轉
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
