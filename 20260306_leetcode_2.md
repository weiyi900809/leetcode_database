leetcode 225 Implement Stack using Queues

class MyStack {
private:
    queue<int> q;

public:
    MyStack() {
        
    }
    
    void push(int x) {
        //  queue是空的
        if(q.empty()){
            q.push(x);
        }
        //  queue不是空
        else{
            int length=q.size();//加入x前的queue大小
            int temp=0;
            q.push(x);
            for(int i=0;i<length;i++ ){
                temp = q.front();
                q.pop();
                q.push(temp);
            }

        }
    }
    
    int pop() {
        int temp;
        temp = q.front();
        q.pop();
        return temp;
    }
    
    int top() {
        return q.front();
    }
    
    bool empty() {
        return q.empty();
    }
};

/**
 * Your MyStack object will be instantiated and called as such:
 * MyStack* obj = new MyStack();
 * obj->push(x);
 * int param_2 = obj->pop();
 * int param_3 = obj->top();
 * bool param_4 = obj->empty();
 */
 
 
 一樣經過gemini 的提示 
 他建議我用單一queue 來建構stack
 
 然後我也叫gemini給我一點演算法的提示
 他說要push 新的變數x進stack (FIFO構建的stack)的話
 push x之後將之前已經在FIFO的變數pop出來並push會FIFO 直到碰到X為止
 舉例來說 FIFO 原本有1 2 現在要push 3
 push 3 之後 FIFO 的狀態為： 3 2 1 由左到右是時間越早 也就是1最早被push 進FIFO 
 接下來將1 pop 出來 再push 進FIFO 接著再將2 pop 出來 再push 進FIFO
 如此一來 FIFO的狀態就會變成 2 1 3 
 這樣呼叫queue的top 就是3 也就是這一次push 的值 
 呼叫queue的pop 也是直接pop 3
