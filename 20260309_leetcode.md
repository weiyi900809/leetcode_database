leetcode 190 Reverse Bits


class Solution {
public:
    int reverseBits(int n) {
        int result=0;
        for(int i=0;i<32;i++){//這裡我原本用while(n!=0) 但不行 因為要完整跑完32bit 
            result = result << 1;
            result = result | (n & 1);//這裡我一開始寫 先寫入再shift 但這樣會錯 因為最後一round 的shift會讓結果整個錯掉
            n = n >> 1;
        }
        return result;
    }
};

leetcode 268 Missing Number

class Solution {
public:
    int missingNumber(vector<int>& nums) {
        int result = nums.size();
        int size=result;
        for(int i=0;i<size;i++){
            result^=i ^ nums[i]; //原本寫 result^= nums[i] 但是這樣不會全部抵消 要將index 也丟進去才行
            cout<<result<<endl;
        }
        return result;
    }
};
