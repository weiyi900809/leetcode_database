682. Baseball Game

class Solution {
public:
    int calPoints(vector<string>& operations) {
        // 1. 嚴謹宣告：計分板必須裝「整數 (int)」，才能應付負數和多位數
        stack<int> st; 
        
        // 2. 迴圈讀取：從陣列裡拿出「字串 (string)」
        for(string op : operations){
            // 注意：字串的比對必須用雙引號 ""
            if(op == "C" && !st.empty()){
                st.pop();
            }
            else if(op == "D"){
                // 因為裡面已經是整數了，直接拿出來乘 2 就好，不用再減 '0'
                int temp = st.top();
                st.push(temp * 2);
            }
            else if(op == "+"){
                // 這裡完美保留你推導出的連招邏輯！
                int temp1 = st.top();
                st.pop();
                int temp2 = st.top();
                
                st.push(temp1);          // 把第一個數字放回去
                st.push(temp1 + temp2);  // 把兩個數字相加的結果放進去
            }
            else{
                // 如果不是上面三種指令，代表它是一個「數字字串」 (例如 "-2" 或 "5")
                // 使用 C++ 內建的 stoi() 將字串轉換為真正的整數，然後 Push 進去
                st.push(stoi(op));
            }
        }

        // 3. 結算：題目要求回傳「計分板上所有數字的總和」
        int sum = 0;
        while(!st.empty()){
            sum += st.top(); // 把最上面的數字加進總和
            st.pop();        // 拿掉剛剛加過的數字
        }
        
        return sum; // 回傳最終得分
    }
};
