733. Flood Fill


class Solution {
public:
    // 裝備背包齊全的 DFS 病毒函式
    void dfs_color(vector<vector<int>>& image, int r, int c, int originalColor, int newColor){
        
        // ==========================================
        // 🛡️ 三層防呆安檢 (順序絕對不能錯！)
        // ==========================================
        
        // 🚨 坑點 1：無限迴圈陷阱
        // 如果起點的顏色剛好等於要染的新顏色，直接提早下班。
        // 否則病毒會以為「顏色對了！繼續染！」，導致原地無限互踢 (Stack Overflow)。
        if( originalColor == newColor ) return;
        
        // 🚨 坑點 2：安檢順序的致命地雷 (Out of Bounds)
        // 【絕對要先看】自己有沒有跌出地圖邊界！
        // 如果 r 已經是 -1，你還先去查 image[-1][c] 的顏色，程式會當場崩潰 (Segmentation Fault)。
        if( r < 0  || c < 0 || r >= image.size() || c >= image[0].size()) return;
        
        // 🚨 坑點 3：顏色防護罩 (順便充當 Visited 標記)
        // 確定人還在地圖裡，才檢查這格的顏色是不是「通緝令」上的目標。
        // 如果這格已經被染成 newColor 了，這行也會把它擋下來，防止走回頭路！
        if( image[r][c] != originalColor ) return;
        
        
        // ==========================================
        // 🦠 感染與擴散
        // ==========================================
        
        // 把這格染成新顏色 (這步就等於幫這個座標打上「已拜訪」標記)
        image[r][c] = newColor;

        // 往四面八方派出新的病毒 (DFS 遞迴)
        dfs_color(image, r - 1, c, originalColor, newColor); // 往上
        dfs_color(image, r + 1, c, originalColor, newColor); // 往下
        dfs_color(image, r, c - 1, originalColor, newColor); // 往左
        dfs_color(image, r, c + 1, originalColor, newColor); // 往右
    }
    
    vector<vector<int>> floodFill(vector<vector<int>>& image, int sr, int sc, int color) {
        
        // 先幫起點拍張照，這就是傳遞給病毒的「通緝令」
        int oriColor = image[sr][sc];
        
        // 投下第一顆病毒，開始從起點 (sr, sc) 擴散
        dfs_color(image, sr, sc, oriColor, color);
        
        // 擴散完畢，回傳整張被塗改過的地圖
        return image;
    }
};
