222. Count Complete Tree Nodes


/**
 * Definition for a binary tree node.
 * struct TreeNode {
 * int val;
 * TreeNode *left;
 * TreeNode *right;
 * TreeNode() : val(0), left(nullptr), right(nullptr) {}
 * TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 * TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int countNodes(TreeNode* root) {
        
        // 🚨 終止條件：如果節點為空，代表沒有節點，回傳 0 (Base Case)
        if(root == nullptr ) return 0;

        // ==========================================
        // 🕵️‍♂️ 階段 1：探測當前子樹的「極左深度」與「極右深度」
        // ==========================================
        TreeNode* temp = root;
        int leftdepth = 0;
        int rightdepth = 0;
        
        // 探險家 1 號：一路向左衝到底，計算左邊界深度
        while( temp != nullptr ){
            leftdepth++;
            temp = temp -> left;
        }
        
        // 探險家 2 號：回到起點，一路向右衝到底，計算右邊界深度
        temp = root;
        while( temp != nullptr ){
            rightdepth++;
            temp = temp -> right;
        }

        // ==========================================
        // 🚀 階段 2：抄捷徑 (Guard Clause)
        // ==========================================
        // 💡 核心優化：如果極左深度 == 極右深度，代表這是一棵「滿二元樹 (Perfect Binary Tree)」！
        // 滿二元樹的節點總數公式為 2^h - 1。
        // 我們利用 C++ 的位元左移運算 `1 << rightdepth` (等同於 2 的 depth 次方) 
        // 只需要一個 CPU 指令就能極速算出答案，提早結束遞迴！
        if( rightdepth == leftdepth ) {
            return (1 << rightdepth) - 1;
        }
        
        // ==========================================
        // 🧱 階段 3：普通遞迴拆解 (O(log^2 N) 的秘密)
        // ==========================================
        // 如果左右深度不同，代表底層沒滿，這是一棵普通的「完全二元樹」。
        // 我們乖乖把它拆成「自己(1) + 左子樹總和 + 右子樹總和」。
        // (💡 神奇的數學特性：完全二元樹的左右子樹中，【必定】有一棵會是滿二元樹，
        // 所以下一次往下遞迴時，絕對會觸發上方的捷徑，大幅砍掉運算時間！)
        else {
            return 1 + countNodes( root -> right) + countNodes( root -> left);
        }
    }
};
