100. Same Tree

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 * int val;
 * TreeNode *left;
 * TreeNode *right;
 * TreeNode() : val(0), left(nullptr), right(nullptr) {}
 * TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 * TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool isSameTree(TreeNode* p, TreeNode* q) {
        // ==========================================
        // 1. 嚴謹的邊界防禦 (Base Case)
        // ==========================================
        // 情境 A：兩邊的主管都走到一塊空地 (沒有子節點了)
        // 既然都沒有東西，代表這部分的「虛無結構」是一模一樣的，回報 true。
        if (p == nullptr && q == nullptr) {
            return true;
        }
        
        // 情境 B：程式能走到這裡，代表 p 和 q「絕對不可能同時是空的」。
        // 此時如果還有其中一個是空地 (另一個必定有房子)，代表兩邊的「結構不對稱」！
        // 直接判定為不一樣，回報 false。
        if (q == nullptr || p == nullptr) {
            return false;
        }
        
        // ==========================================
        // 2. 收集情報 (Recursive Calls)
        // ==========================================
        // 程式走到這裡，保證 p 和 q 都真有其人 (都有實體節點)。
        // 長官下令：分別去調查兩家公司的「左部門」和「右部門」是不是長得一模一樣。
        bool left_child_same = isSameTree(p->left, q->left);
        bool right_child_same = isSameTree(p->right, q->right);
        
        // ==========================================
        // 3. 統整並回報 (Return Result)
        // ==========================================
        // 兩家公司要完全一樣，必須「同時滿足」以下三個嚴苛條件：
        // 1. 目前這兩個長官自己的業績 (數值) 一樣
        // 2. 且 (&&) 左部門回報的結果是一樣的
        // 3. 且 (&&) 右部門回報的結果是一樣的
        return (p->val == q->val) && left_child_same && right_child_same;
    }
};
