739. Daily Temperatures

class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        int arr_size = temperatures.size();
        
        // 【防禦性設計】將結果陣列預設全部填滿 0。
        // 這樣如果某些日子直到最後都等不到升溫，它們的答案自然就是 0，不需要額外處理。
        vector<int> result(arr_size, 0); 
        
        // 單調堆疊：我們的「懸案檔案櫃」，裡面專門存放【日期的 Index】
        stack<int> s;

        for (int i = 0; i < arr_size; i++) {
            int curr_temp = temperatures[i];
            
            // ==========================================
            // 觸發結算機制 (連鎖 Pop)
            // ==========================================
            // 條件 1：堆疊不能是空的 (防止 s.top() 存取越界)
            // 條件 2：今天的溫度，比檔案櫃最上面那天的溫度還要高 (破案了！)
            while (!s.empty() && curr_temp > temperatures[s.top()]) {
                
                // 1. 拿出頂部的懸案 (過去的某一天)
                int prev_day_index = s.top();
                
                // 2. 計算等待的天數 (今天 - 過去那天)，寫入答案卷
                result[prev_day_index] = i - prev_day_index; 
                
                // 3. 結案，把它從櫃子裡徹底移除
                s.pop();
            }
            
            // ==========================================
            // 歸檔等待 (Push)
            // ==========================================
            // 不管今天有沒有幫過去的日子破案，今天自己也需要等待未來的升溫，
            // 所以一定要把「今天的 Index」丟進檔案櫃。
            s.push(i);
        }
        
        return result;
    }
};
