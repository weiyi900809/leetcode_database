56. Merge Intervals


class Solution {
public:
    vector<vector<int>> merge(vector<vector<int>>& intervals) {
        // 防呆機制
        if (intervals.empty()) return {};
        
        // 依照開始時間排序
        sort(intervals.begin(), intervals.end(), [](const vector<int>& a, const vector<int>& b) {
            return a[0] < b[0]; 
        });
        
        vector<vector<int>> ans;
        
        // curr 專門用來記錄「目前 ans 陣列中，最後一個元素的 index」
        int curr = 0; 
        
        // 起手式：把第 0 個會議丟進背包
        ans.push_back(intervals[0]);
        
        for (int i = 1; i < intervals.size(); i++) {
            
            // 拷問：新會議的開始時間，有沒有 <= 背包最後一個會議的結束時間？
            if (intervals[i][0] <= ans[curr][1]) {
                
                // ==========================================
                // 🚨 坑點 1：大魚吃小魚 (忘記用 max)
                // ==========================================
                // 如果原本是 [1, 5]，遇到 [2, 3]，如果不取 max，
                // 直接寫 ans[curr][1] = intervals[i][1]，會議就會縮水成 [1, 3]！
                ans[curr][1] = max(ans[curr][1], intervals[i][1]);
                
            } 
            else {
                
                // ==========================================
                // 🚨 坑點 2：搞混了原始陣列與答案陣列的 index
                // ==========================================
                // 把新的獨立會議丟進背包裡。
                ans.push_back(intervals[i]);
                
                // 因為背包裡面多了一個元素，所以代表「最後一個位置」的 curr 也要往後走一步。
                // 絕對不能寫 curr = i，因為 i 走得比 curr 快很多！
                curr++; 
            }
        }
        
        return ans;
    }
};
