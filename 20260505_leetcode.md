3. Longest Substring Without Repeating Characters

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        // 準備裝備：雙指標與最大長度記錄器
        int left = 0, right = 0; 
        int max_len = 0;

        // 胃袋：用來以 O(1) 的極速檢查字元是否重複
        unordered_set<char> window; 
        
        // 讓毛毛蟲的頭部 (right) 一路往右吃到字串結尾
        while (right < s.length()) {
            
            // ==========================================
            // 🚨 坑點 1：必須使用 while，絕對不能用 if！
            // ==========================================
            // 初學者常犯的錯是寫 if(window.count(s[right]))。
            // 但重複的字元可能卡在胃袋「中間」而不是剛好在「尾巴」。
            // 必須用 while 逼迫 left 不斷往前縮，連同前面的無辜字元一起吐掉，
            // 直到警報解除 (胃袋裡完全沒有 s[right]) 為止。
            while (window.count(s[right])) {
                
                // ==========================================
                // 🚨 坑點 2：刪除目標是 s[left]，千萬別刪錯！
                // ==========================================
                // 腦袋不清楚時很容易錯寫成 window.erase(s[right])，這會導致無窮迴圈。
                // 記住物理意義：吐出來的永遠是「尾巴 (left)」吃進去的舊東西。
                window.erase(s[left]); // 尾巴開始吐
                left++;                // 尾巴往前縮
            }
            
            // 警報解除！現在確定胃袋裡沒有重複字元了，放心把新字元吃下去
            window.insert(s[right]);
            
            // ==========================================
            // 🚨 坑點 3：長度計算的時機與公式
            // ==========================================
            // 區間 [left, right] 的元素總數公式是：right - left + 1。
            // 這個計算「必須」放在 right++ 之前執行，才能精準算出當下毛毛蟲的真實長度。
            max_len = max(max_len, right - left + 1);
            
            // 結算完畢，頭部繼續往前伸，準備吃下一個字元
            right++;
        }
        
        return max_len;
    }
};
