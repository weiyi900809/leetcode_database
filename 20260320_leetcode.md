146. LRU Cache


class LRUCache {
public:
    // 雙向鏈結串列：用來排隊。裡面存 pair<int, int> 分別代表 {key, value}
    // 為什麼要存 key？因為當cache滿了，要刪除tail元素時，需要這個 key 去 Map 裡把紀錄清掉
    typedef pair<int, int> key_value_pair;
    typedef list<key_value_pair>::iterator list_iter;
    list<key_value_pair> cache_list;
    
    // Hash Map：Key 是題目的 key，Value 是那筆資料在 list 中的「迭代器 (iterator / 記憶體位置)」
    unordered_map<int, list_iter> cache_map;

    int cache_capacity=0;
    LRUCache(int capacity) {
        cache_capacity=capacity;
    }
    
    int get(int key) {
        auto it = cache_map.find(key);
        if(it == cache_map.end()){
            return -1;
        }
        else{
            cache_list.splice(cache_list.begin(), cache_list, it->second);//最近被使用
            /* splice的 三個parameter 的意思分別是
            參數一：cache_list.begin() (目標位置)      意義：你要把節點「貼」到哪裡？
            參數二：cache_list (來源串列) 意義：你要剪下來的節點，原本在哪個串列裡？
            參數三：it->second (目標節點)  意義：你到底要「剪」哪一個節點？請給我它的位置（Iterator）。
            */
            return it->second->second;
        }
    }
    
    void put(int key, int value) {
        // 第一步：先去 Map 裡找找看這個 key 存不存在
        auto it = cache_map.find(key);

        // 情況 A：這個 key 已經存在 (這只是個更新動作)
        if (it != cache_map.end()) {
            // 1. 既然找到了，先更新它的值 (它在 List 裡的 value)
            // 提示：你可以透過 it->second (拿到 List iterator) 再透過 ->second 拿到並修改 value
            it->second->second = value;
            // 2. 因為它剛被存取，所以要把它搬到 List 的最前面 (最新鮮的位置)
            cache_list.splice(cache_list.begin(), cache_list, it->second);
            return; // 更新完就結束了，提早 return 離開
        }

        // 情況 B：這個 key 不存在，這是一筆全新的資料！
        // 第二步：因為要加新資料了，必須先檢查「客滿了沒？」
        if (cache_list.size() == cache_capacity) {
            // 如果客滿了，必須踢掉最久沒被用的那筆 (在 List 的最後面)
        
            // 1. 拿到最尾端那個節點的 key (因為等一下要去 Map 裡刪除它)
            int last_item_key=cache_list.back().first;
            // 2. 把這個舊的 key 從 Hash Map 裡面移除
            cache_map.erase(last_item_key);
            // 3. 把最尾端的節點從 List 裡面刪除
            cache_list.pop_back();
        }

        // 第三步：現在確定有空間了，可以把新資料加進來了！
        
        // 1. 把新的 {key, value} 塞入 List 的最前面

        cache_list.push_front({key, value});
        // 2. 最重要的一步：在 Hash Map 裡記錄這筆新資料在 List 裡的位置
        // 提示：它剛剛被塞到最前面，所以它的位置就是 cache_list.begin()
        cache_map[key] = cache_list.begin();
    }
};

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache* obj = new LRUCache(capacity);
 * int param_1 = obj->get(key);
 * obj->put(key,value);
 */
