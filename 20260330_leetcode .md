700. Search in a Binary Search Tree


/**
 * Definition for a binary tree node.
 * struct TreeNode {
 * int val;
 * TreeNode *left;
 * TreeNode *right;
 * TreeNode() : val(0), left(nullptr), right(nullptr) {}
 * TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 * TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* searchBST(TreeNode* root, int val) {
        // ==========================================
        // 【嚴謹的 C++ 記憶體防禦：指標初始化】
        // ==========================================
        // 🚨 這是 C++ 非常重要的核心觀念！
        // 如果只寫 `TreeNode* leftside;`，裡面裝的會是不可預期的「垃圾記憶體位置」。
        // 當後面的程式沒有機會修改它 (例如 if 條件沒成立)，最後又把它 return 出去時，
        // 系統一讀取這個垃圾位置，就會發生 Segmentation Fault (記憶體區段錯誤) 導致當機。
        // 所以，宣告指標時「務必」要立刻賦予 nullptr 初始化，確保它的初始狀態是安全的「空」。
        TreeNode* leftside = nullptr;
        TreeNode* rightside = nullptr;

        // ==========================================
        // 1. 邊界防禦與成功條件 (Base Cases)
        // ==========================================
        // 如果總裁點到一塊空地 (走到死胡同)，代表整條路都沒這個人，回報空值。
        if(root == nullptr){
            return nullptr;
        }
        
        // 恭喜！總裁的數字剛好就是我們要找的目標，直接把總裁 (和他的底下部門) 回報上去。
        if(root->val == val){
            return root;
        }

        // ==========================================
        // 2. 核心決策：依據 BST 的排序規矩交辦任務
        // ==========================================
        // 規矩 A：如果總裁的數字「大於」目標值 (代表目標比較小)
        // 那目標絕對只可能藏在「左邊」的部門裡。
        if(root->val > val){
            // 派左副總去找，並把他的尋找結果 (可能找到，也可能是 nullptr) 暫存在 leftside
            leftside = searchBST(root->left, val);
        }
        
        // 規矩 B：如果總裁的數字「小於」目標值 (代表目標比較大)
        // 那目標絕對只可能藏在「右邊」的部門裡。
        if(root->val < val){
            // 派右副總去找，並把他的尋找結果暫存在 rightside
            rightside = searchBST(root->right, val);
        }
        
        // ==========================================
        // 3. 統整並回報 (Ternary Operator)
        // ==========================================
        // 白話文解釋這個三元運算子：
        // 「如果左邊部門回報『找不到人 (nullptr)』，那我就死馬當活馬醫，把右邊部門的回報交上去；
        // 反之，如果左邊有找到人，我就直接交上左邊的結果。」
        // 
        // 💡 雖然在 BST 的嚴格規矩下，左右兩邊其實只會有一邊被實際派出去尋找 (另一個必定保持為 nullptr)，
        // 但這個寫法確實能涵蓋最後的邏輯，把唯一有動作的那一邊的結果正確回傳。
        return (leftside == nullptr) ? rightside : leftside;
    }
};
