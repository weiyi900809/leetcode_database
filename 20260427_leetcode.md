724. Find Pivot Index


class Solution {
public:
    int pivotIndex(vector<int>& nums) {
        // ==========================================
        // 📍 變數準備區
        // ==========================================
        int TotalSum = 0; // 記錄整條陣列的總和
        int LeftSum = 0;  // 記錄「目前中心點」左邊的所有數字總和
        int RightSum = 0; // 記錄「目前中心點」右邊的所有數字總和
        
        // 第一圈：先花 O(N) 的時間，把整條陣列的總和算出來
        for(int i = 0; i < nums.size(); i++){
            TotalSum += nums[i];
        }
        
        // ==========================================
        // 🚀 前綴和核心邏輯：邊走邊扣除
        // ==========================================
        // 第二圈：由左往右，把每個數字都當作「可能的中心點 (Pivot)」試試看
        for(int i = 0; i < nums.size(); i++){
            
            // 💡 核心魔術：不需要再跑迴圈算右邊！
            // 根據公式：整條總和 - 左邊總和 - 中心點自己 = 右邊總和
            RightSum = TotalSum - LeftSum - nums[i];
            
            // 🎯 判斷是否命中：如果左邊等於右邊，這個 index 就是完美的中心點
            if(RightSum == LeftSum) {
                return i;
            }
            
            // 🏃‍♂️ 狀態推進 (交接棒)：
            // 既然當下這個數字不是中心點，那我們要繼續往下一個數字前進了。
            // 所以「現在這個數字」，就會變成「下一個人的左半邊」的一部分！
            // 記得把它加進 LeftSum 裡。
            LeftSum += nums[i];
        }
        
        // 如果整個陣列都走完了還是找不到，依題目規定回傳 -1
        return -1;
    }
};
