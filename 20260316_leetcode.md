88. Merge Sorted Array

class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        // 1. 初始化三個讀寫指標 (從尾巴開始逆向操作)
        int nums1_ptr = m - 1;                // 指向 nums1 有效資料的最高位
        int nums2_ptr = n - 1;                // 指向 nums2 有效資料的最高位
        int write_ptr = nums1.size() - 1;     // 指向 nums1 實體陣列的最後一格空位 (Buffer 尾端)
        
        // 2. 雙指標逆向比對與搬移
        // 只要兩個陣列都還有資料，就挑比較大的寫進 write_ptr
        while(nums2_ptr >= 0 && nums1_ptr >= 0){
            if(nums2[nums2_ptr] > nums1[nums1_ptr]){
                nums1[write_ptr] = nums2[nums2_ptr]; // nums2 的數字較大，寫入
                nums2_ptr--;                         // nums2 指標退一格
            }
            else{
                nums1[write_ptr] = nums1[nums1_ptr]; // nums1 的數字較大，寫入
                nums1_ptr--;                         // nums1 指標退一格
            }
            write_ptr--; // 寫入完成，寫入點往前移一格準備下一次寫入
        }
        
        // 3. 邊界收尾 (Edge Case Handling)
        // 情況 A：如果 nums2 先搬完 (nums2_ptr < 0)，代表 nums1 剩下的資料本來就在正確位置，不須處理。
        // 情況 B：如果 nums1 先搬完 (nums1_ptr < 0)，但 nums2 還有剩，必須把 nums2 剩下的資料填入 nums1 前面。
        while(nums2_ptr >= 0){
            nums1[write_ptr] = nums2[nums2_ptr];
            nums2_ptr--;
            write_ptr--;
        }
    }
};

141. Linked List Cycle

class Solution {
public:
    bool hasCycle(ListNode *head) {
        // 1. 初始化快慢指標，起點都在 head
        ListNode *slow = head;
        ListNode *fast = head;
        
        // 2. 邊界防呆：偵察機 fast 必須確保「自己」與「下一格」都不是 NULL，
        // 才能安全執行後面的 fast->next->next (踩兩步)
        while(fast != NULL && fast->next != NULL){
            
            // 烏龜走一步，兔子走兩步
            slow = slow->next;
            fast = fast->next->next;
            
            // 3. 狀態機崩潰偵測：如果在某個 Clock 週期，兔子和烏龜在同一個記憶體位址相遇了，
            // 代表他們陷入了無窮迴圈 (Cycle)
            if(fast == slow){
                return true;
            }
        }
        
        // 4. 順利脫出迴圈，代表偵察機 fast 碰到了 NULL 終點，證明這是一條沒有環的單行道
        return false;
    }
};
