70. Climbing Stairs


class Solution {
public:
    int climbStairs(int n) {
        // 🛡️ 保鑣：1 階和 2 階直接回傳
        if (n <= 2) return n;
        
        // 📍 變數命名優化：讓數字對應它的「物理意義」
        int prev2 = 1; // 代表退 2 步 (n-2) 的方法數
        int prev1 = 2; // 代表退 1 步 (n-1) 的方法數
        
        for (int i = 3; i <= n; i++) {
            // ✨ 優化：讓 current (也就是你的 result) 在迴圈內出生，用完即丟！
            int current = prev1 + prev2; 
            
            // 🏃‍♂️ 交接棒時間 (大家往前走一格)
            prev2 = prev1;   // 原本的 n-1 變成了下一步的 n-2
            prev1 = current; // 剛剛算出的新答案，變成了下一步的 n-1
        }
        
        // 迴圈結束時，prev1 剛好就會握著最後一階的答案！
        return prev1; 
    }
};
