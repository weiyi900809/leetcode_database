108. Convert Sorted Array to Binary Search Tree


/**
 * Definition for a binary tree node.
 * struct TreeNode {
 * int val;
 * TreeNode *left;
 * TreeNode *right;
 * TreeNode() : val(0), left(nullptr), right(nullptr) {}
 * TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 * TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    // 這是你寫的超強內部工具：全自動部門建構機
    TreeNode* buildTree(vector<int>& nums, int left, int right){
        // 1. 邊界防禦：如果名單被切到空了 (左邊界跑到右邊界右邊)，回報空地
        if(left > right){
            return nullptr;
        }
        
        // 2. 找出這份名單的正中間 (中位數)，提拔為這個部門的主管
        int mid = left + (right - left) / 2;
        TreeNode* node = new TreeNode(nums[mid]);
        
        // 3. 遞迴交辦：
        // 把主管左邊的名單 (left 到 mid-1) 交下去建左部門
        node->left = buildTree(nums, left, mid - 1);
        // 把主管右邊的名單 (mid+1 到 right) 交下去建右部門
        node->right = buildTree(nums, mid + 1, right);
        
        // 回報建好的部門主管
        return node;
    }
    
    // 這是 LeetCode 呼叫的主任務 (總裁辦公室)
    TreeNode* sortedArrayToBST(vector<int>& nums) {
        // 總裁不需要自己動手！
        // 直接啟動自動建構機，給定整份名單的範圍：從 0 到 nums.size() - 1
        // 機器就會自動蓋好整棵平衡樹，並回傳最高總裁 (root) 給你
        return buildTree(nums, 0, nums.size() - 1);
    }
};
