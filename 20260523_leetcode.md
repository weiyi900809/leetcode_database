994. Rotting Oranges



class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int fresh_orange = 0;
        queue<pair<int,int>> q; 
        int m = grid.size();
        int n = grid[0].size();
        
        // ==========================================
        // 階段 1：全局掃描 (Initialization)
        // ==========================================
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                if(grid[i][j] == 1){
                    fresh_orange++; // 統計存活的新鮮橘子數量
                }
                else if(grid[i][j] == 2){
                    q.push({i, j}); // 徵召所有初始感染源
                }
            }
        }
        
        // 【邊界防禦】如果一開始就沒有新鮮橘子，無論有沒有爛橘子，都需要 0 分鐘
        if(fresh_orange == 0) return 0;
        
        int min = 0;
        
        // 【方向陣列 Direction Array】
        // 業界標準寫法：分別對應 上(-1,0), 下(1,0), 左(0,-1), 右(0,1)
        int dx[4] = {-1, 1, 0, 0};
        int dy[4] = {0, 0, -1, 1};
        
        // ==========================================
        // 階段 2：時脈驅動模擬 (Clock-Cycle Simulation)
        // ==========================================
        // 【優化條件】當 queue 內還有感染源，且 "還有新鮮橘子" 時才繼續計時
        while(!q.empty() && fresh_orange > 0){
            // 鎖定當前這 1 分鐘內，具備傳染能力的所有爛橘子數量
            int curr_rotten_num = q.size();
            
            // 批次處理這一個時脈週期
            for(int i = 0; i < curr_rotten_num; i++){
                pair<int, int> now = q.front(); 
                q.pop(); // 【關鍵】讀取完務必移出 Queue
                
                int now_x = now.first;
                int now_y = now.second;
                
                // 向 4 個方向發送感染訊號
                for(int d = 0; d < 4; d++){
                    int next_x = now_x + dx[d];
                    int next_y = now_y + dy[d];
                    
                    // 【安全防禦順序】
                    // 絕對必須先檢查邊界 (>=0 且 <m/n)，才能去讀取 grid 的值
                    if(next_x >= 0 && next_x < m && next_y >= 0 && next_y < n && grid[next_x][next_y] == 1){
                        grid[next_x][next_y] = 2; // 感染它
                        fresh_orange--;           // 倖存者減少
                        q.push({next_x, next_y}); // 將新感染的橘子排入下一輪 (下一分鐘) 的 Queue
                    }
                }
            }
            min++; // 這一批次處理完，時間推進 1 分鐘
        }
        
        // ==========================================
        // 階段 3：最終驗收 (Final Verification)
        // ==========================================
        // 如果還有新鮮橘子倖存 (被 0 擋住的孤島)，回傳 -1
        if(fresh_orange > 0){
            return -1;
        }
        
        return min; // 完美全滅，回傳總耗時
    }
};
