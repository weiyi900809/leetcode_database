703. Kth Largest Element in a Stream


class KthLargest {
private:
    // ==========================================
    // 裝備區
    // ==========================================
    int k; // 記錄 VIP 俱樂部的人數上限
    
    // 🚨 坑點 1：Min Heap 的 C++ 專屬咒語
    // 預設的 priority_queue 是 Max Heap（最大值在上面）。
    // 要把它變成 Min Heap，必須乖乖寫完這三個參數，少一個都不行。
    priority_queue<int, vector<int>, greater<int>> min_heap;

    // ==========================================
    // 內部核心邏輯 (Private)
    // ==========================================
    // 🚨 坑點 2：單一職責原則 (Single Responsibility Principle)
    // 故意拆出這個沒有回傳值 (void) 的私有函數，只負責維護 Heap 的狀態。
    // 這樣在 Constructor 迴圈呼叫時，就不會產生「回傳了錯誤的第 K 大數字卻沒人接」的架構瑕疵。
    void insertToHeap(int val) {
        // 動作 A：不管三七二十一，新數字先塞進去
        min_heap.push(val);
        
        // 動作 B：如果俱樂部爆滿了 (超過 K 個人)
        // 🚨 坑點 3：為什麼是 Min Heap？
        // 因為 Min Heap 最上面 (top) 永遠是最弱、最小的數字。
        // 把最弱的踢掉，留在裡面的永遠都是「歷史前 K 強」的數字！
        if (min_heap.size() > k) {
            min_heap.pop();
        }
    }

public:
    // ==========================================
    // 建構子 (初始化)
    // ==========================================
    KthLargest(int k, vector<int>& nums) {
        this->k = k; 
        
        // 初始化時，安心地呼叫內部邏輯，不會有任何多餘的 Side Effect (副作用)
        for(int num : nums) {
            insertToHeap(num);
        }
    }
    
    // ==========================================
    // 對外開放的 API (Public)
    // ==========================================
    int add(int val) {
        
        // 1. 先讓新數字去挑戰俱樂部
        insertToHeap(val);
        
        // 2. 嚴謹的防呆機制 (Defensive Programming)
        // 🚨 坑點 4：資料量不足的致命邊界條件 (Edge Case)
        // 雖然 LeetCode 題目條件用數學式 (nums.length >= k - 1) 保證了不會發生這件事。
        // 但在真實世界的系統中，如果 Heap 裡面不到 K 個人卻呼叫 top()，
        // 拿到的絕對不是真正的「第 K 大」，甚至可能導致程式 Crash。
        if (min_heap.size() < k) {
            return -1; // 實務上也可以 throw runtime_error("Not enough elements");
        }
        
        // 3. 安全交付答案
        // 經過防呆檢查，這裡 100% 確定有至少 K 個人，且 top() 就是第 K 大。
        return min_heap.top();
    }
};
