543. Diameter of Binary Tree



/**
 * Definition for a binary tree node.
 * struct TreeNode {
 * int val;
 * TreeNode *left;
 * TreeNode *right;
 * TreeNode() : val(0), left(nullptr), right(nullptr) {}
 * TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 * TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
private: 
    // ==========================================
    // 🚨 坑點 1：忘記全域變數 (Global State) 的必要性
    // ==========================================
    // 初學者常犯的錯是只在根節點算左加右，以為最長路徑一定會經過 Root。
    // 但最長路徑 (倒 V 字型) 可能完全藏在某個左子樹的深處！
    // 所以我們必須在外面掛一個「全域計分板」，讓任何一個節點只要發現更長的直徑，
    // 就能越級回報，偷偷把它換掉。
    int max_diameter = 0;    

public:
    // ==========================================
    // 核心概念：路徑的「折返點」與「雙重任務」
    // ==========================================
    // 在樹裡面，任何一條路徑一定長得像一個「倒 V 字型」（或直線）。
    // dfs 函式的設計是一心二用：
    // 任務 A (為全域爭光)：把目前節點當成 V 字型的最高點，結算「左深 + 右深」，去挑戰 max_diameter。
    // 任務 B (對上層交差)：回傳「以自己為起點往下走的最大深度」，讓上面的爸爸繼續算。
    int dfs(TreeNode* node) {
        
        // 防禦邊界：探底了，深度為 0
        if(node == nullptr){
            return 0;
        }
        
        // 由下而上 (Bottom-up)：先派人去把左右子樹的深度探勘回來
        int left_depth = dfs(node->left);
        int right_depth = dfs(node->right);
        
        // 任務 A：挑戰全域最高紀錄！
        // 「你左手邊最深能探到多深」 + 「你右手邊最深能探到多深」 = 穿過你的最大直徑
        max_diameter = max(left_depth + right_depth, max_diameter);
        
        // ==========================================
        // 🚨 坑點 2：遞迴回傳值的混淆
        // ==========================================
        // 這裡絕對不能回傳 max_diameter！
        // 對於上一層的爸爸來說，他只想知道「如果往你這邊走，最深可以走多深？」
        // 所以你只能挑左邊或右邊比較深的那一條路，加上你自己這一層 (+1) 回傳上去。
        return max(left_depth, right_depth) + 1; 
    }
    
    int diameterOfBinaryTree(TreeNode* root) {
        // 主帥啟動遞迴引擎
        dfs(root);
        // 引擎跑完後，所有的節點都已經挑戰過全域計分板了，直接回傳結果
        return max_diameter;
    }
};
