278. First Bad Version

// The API isBadVersion is defined for you.
// bool isBadVersion(int version);

class Solution {
public:
    int firstBadVersion(int n) {
        int left = 1;
        int right = n;
        
        // ==========================================
        // 🚨 坑點 1：捕鼠籠模式的邊界條件 (無限迴圈地雷)
        // ==========================================
        // 尋找「狀態反轉的邊界」時，必須使用 < 而不是 <=。
        // 因為當我們把嫌疑犯縮小到只剩最後 1 個時 (left == right)，答案就已經確定了。
        // 如果加上 =，且配合下方的 right = mid，會導致程式在 left == right 的當下，
        // 不斷重複執行 right = mid，陷入時空無限迴圈 (Time Limit Exceeded)！
        while(left < right){
            
            // ==========================================
            // 🚨 坑點 2：整數溢位 (Integer Overflow) 的定時炸彈
            // ==========================================
            // 絕對不能寫成 int mid = (left + right) / 2;
            // 當 left 和 right 都逼近 32-bit int 的極限值 (約 21 億) 時，相加會直接撐爆記憶體變成負數。
            // 改成這種「起點 + 距離的一半」的位移寫法，數學結果完全一樣，但保證絕對不溢位。
            int mid = left + (right - left) / 2;
            
            
            // 呼叫 API 檢查這個切出來的中間版本
            if(isBadVersion(mid) != true){ 
                // 情境 A：mid 是好的 (Good)
                // 代表「第一個壞版本」絕對還在更右邊，而且絕不可能是 mid 自己。
                // 策略：大膽丟棄，左邊界毫不留情地跨過 mid。
                left = mid + 1;
            }
            else {
                // 情境 B：mid 是壞的 (Bad)
                // 代表 mid「可能」就是第一個壞的，但也可能壞掉的源頭在更左邊。
                // 策略：絕不能丟棄 mid！把右邊界拉到 mid，將它關在捕鼠籠裡繼續調查。
                right = mid;
            }
        }
        
        // 當迴圈結束的瞬間，left 必定等於 right。
        // 區間被夾擠到只剩一個數字，這個數字 100% 就是我們要找的「第一個壞版本」。
        return left;
    }
};
