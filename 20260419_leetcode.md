637. Average of Levels in Binary Tree


/**
 * Definition for a binary tree node.
 * struct TreeNode {
 * int val;
 * TreeNode *left;
 * TreeNode *right;
 * TreeNode() : val(0), left(nullptr), right(nullptr) {}
 * TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 * TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<double> averageOfLevels(TreeNode* root) {
        vector<double> result;
        
        // 📦 BFS 的核心武器：FIFO (先進先出) 佇列
        queue<TreeNode*> q; 
        
        // ==========================================
        // 🛡️ 門口保鑣 (Guard Clause)
        // ==========================================
        // 如果是空樹，直接利用現代 C++ (C++11) 的語法糖 `{}` 
        // 讓編譯器自動幫我們回傳一個乾淨的空陣列 vector<double>。
        if (root == nullptr) return {}; 
        
        // 🚩 起手式：把根節點放進佇列，準備開始逐層掃描
        q.push(root); 
        
        // 只要佇列裡面還有人排隊，就代表樹還沒走完
        while(!q.empty()){
            
            // ==========================================
            // 🔒 鎖定本層人數 (BFS 模板最關鍵的一步！)
            // ==========================================
            // 必須在進入小迴圈前，先把現在 Queue 裡的人數 (size) 記下來。
            // 這樣才能精準區分「這一層的人」和等一下會被塞進來的「下一層的小孩」。
            int size = q.size(); 
            
            // 🚨 型態陷阱防禦：
            // 必須宣告為 double，否則最後的 sum / size 會發生整數除法，導致小數點被無情捨去！
            double sum = 0; 
            
            // 🏃‍♂️ 小迴圈：嚴格執行 size 次，專心處理「當前這一層」的所有節點
            for(int i = 0; i < size; i++){
                
                // ✨ Clean Code 微調：將 cur 宣告在迴圈內，縮小生存空間，用完即丟。
                TreeNode* cur = q.front();  // 1. 偷看排在最前面的那個人
                q.pop();                    // 2. 叫他離開隊伍 (出隊)
                
                // ==========================================
                // 👶 托兒所邏輯：把小孩丟到隊伍最後面
                // ==========================================
                // 如果當前節點有左/右小孩，就把他們排進隊伍的最後面。
                // (因為 FIFO 特性，他們會乖乖等到下一輪的大迴圈才被處理)
                if(cur->left != nullptr){
                    q.push(cur->left);
                }
                if(cur->right != nullptr){
                    q.push(cur->right);
                }
                
                // 💰 累加這一層的數值
                sum += cur->val; 
            }
            
            // 🎯 這一層 (小迴圈) 處理完畢，計算平均值並打包出貨！
            result.push_back(sum / size);
        }
        
        return result;
    }
};
