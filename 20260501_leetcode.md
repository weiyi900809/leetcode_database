455. Assign Cookies

class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
        
        // ==========================================
        // 階段 1：貪心前置作業 (Sorting)
        // ==========================================
        
        // 🚨 坑點 1：C++ 專屬的排序語法
        // 不能寫成直覺的 g.sort() (那是 Python / JS 的寫法)。
        // 必須傳入指標 (Iterator)，告訴電腦記憶體的「起點」與「終點」。
        
        sort(g.begin(), g.end()); // 把小孩的胃口由小到大排好 (最容易滿足的優先)
        sort(s.begin(), s.end()); // 把餅乾的尺寸由小到大排好 (最不浪費的優先)
        
        // ==========================================
        // 階段 2：雙指標 (Two Pointers) 貪心匹配
        // ==========================================
        int child_counter = 0;  // 負責指著目前「胃口最小的小孩」
        int cookie_counter = 0; // 負責指著目前「尺寸最小的餅乾」
        
        // 🚨 坑點 2：越界當機防護 (Out of Bounds)
        // 絕對不能只寫 while(child_counter < g.size())！
        // 如果餅乾發完了，cookie_counter 會超出陣列範圍，下一秒去拿 s[cookie_counter] 就會當場崩潰。
        // 必須使用 && 綁定：小孩還在排隊「而且」手邊還有餅乾，才能繼續發。
        while (child_counter < g.size() && cookie_counter < s.size()) {
            
            // 貪心決策：這塊餅乾能滿足這個小孩嗎？
            if (g[child_counter] <= s[cookie_counter]) {
                // 配對成功！
                // 小孩拿到餅乾，開心離開 (換下一個小孩)
                child_counter++;
                // 餅乾被吃掉了 (換下一塊餅乾)
                cookie_counter++;
            } 
            else {
                // 配對失敗：餅乾太小了！
                // 這塊餅乾連目前胃口最小的小孩都餵不飽，留著也沒用，直接作廢。
                // 小孩留在原地繼續等，我們換下一塊「更大」的餅乾試試看。
                cookie_counter++;
            }
        }
        
        // ==========================================
        // 階段 3：結算輸出
        // ==========================================
        // 因為小孩的指標 (child_counter) 是從 0 開始的，
        // 他每往前走一步，就代表有一個小孩被餵飽了。
        // 所以迴圈結束時，指標停留的數字，剛好就是成功餵飽的小孩總數！
        return child_counter;
    }
};
