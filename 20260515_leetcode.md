387. First Unique Character in a String

class Solution {
public:
    int firstUniqChar(string s) {
        
        // ==========================================
        // 核心武器：極速字元計數器 (陣列模擬 Hash Map)
        // ==========================================
        // 宣告一個長度 26 的陣列，初始值皆為 0。
        // 利用 ASCII 碼的連續性，把 'a'~'z' 對應到 index 0~25。
        vector<int> count(26, 0);
        
        // ==========================================
        // 第一趟掃描：人口普查 (Build Frequency Map)
        // ==========================================
        for(const char c : s){
            // 透過減去 'a' 計算出 index，並將該字母的出現次數 +1
            count[c - 'a']++;
        }
        
        // ==========================================
        // 第二趟掃描：依序查核 (Find the Winner)
        // ==========================================
        // 🚨 坑點 1：搞錯掃描對象 (順序地雷)
        // 很多初學者第二趟會去掃描 count 陣列 (i = 0 到 25)。
        // 這樣做會找到「字母表裡排最前面的單身狗 (例如 a)」，而不是「字串裡最先出現的單身狗」。
        // ✅ 正確做法：必須從頭掃描「原始字串」，確保順序正確。
        for(int i = 0; i < s.size(); i++){
            char c = s[i];
            
            // 去計分板查詢這個字母的總出現次數，如果是 1，立刻結案！
            if(count[c - 'a'] == 1){
                return i;
            }
        }
        
        // 如果整條字串掃完都沒有單身狗，回傳 -1
        return -1;
    }
};
