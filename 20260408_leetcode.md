125. Valid Palindrome

class Solution {
public:
    bool isPalindrome(string s) {
        string result="";
        for(int i=0;i<s.size();i++){
            if(s[i] >= 'a' && s[i] <= 'z') {
                result += s[i];
            }
            else if(s[i] >= 'A' && s[i] <= 'Z'){
                result += s[i]+abs('A'-'a');
            }
            else if(s[i] >= '0' && s[i] <= '9'){
                result += s[i];
            }

        }
        int left=0;
        int right=result.size()-1;
        while(left <= right){
            if(result[left] != result[right] ){
                return false;
            }
            left++;
            right--;
        }
        return true;
    }
};

class Solution {
public:
    bool isPalindrome(string s) {
        int left = 0;
        int right = s.size() - 1;

        // 當雙指標還沒相遇時，持續往中間逼近
        while (left < right) {
            
            // 1. 左指標：遇到「非英數字」就跳過
            // 🚨 嚴謹防禦：必須加上 left < right，防止字串全是符號 (如 "   ") 導致指標越界當機
            while (left < right && !isalnum(s[left])) {
                left++;
            }
            
            // 2. 右指標：遇到「非英數字」就跳過
            // 🚨 同樣的嚴謹防禦
            while (left < right && !isalnum(s[right])) {
                right--;
            }
            
            // 3. 執行比對：把雙方都轉成小寫後比對
            if (tolower(s[left]) != tolower(s[right])) {
                return false; // 只要有一對不一樣，就絕對不是迴文
            }
            
            // 4. 比對成功，雙方各自往前/後走一步，準備驗證下一對
            left++;
            right--;
        }
        
        // 全部比對完都沒發現問題，安全過關
        return true;
    }
};

2108. Find First Palindromic String in the Array

class Solution {
public:
    // 輕量版：拿掉多餘的防呆，並使用 const string& 避免複製
    bool isPalindrome(const string& s) {
        int left = 0;
        int right = s.size() - 1;

        while (left < right) {
            if (s[left] != s[right]) {
                return false;
            }
            left++;
            right--;
        }
        return true;
    }

    string firstPalindrome(vector<string>& words) {
        // 業界標準：使用 const string& 走訪陣列，效能最高且語法乾淨
        for (const string& word : words) {
            if (isPalindrome(word)) {
                return word;
            }
        }
        
        return "";
    }
};
