36. Valid Sudoku

這題難在要怎麼找到規律 導出九宮格編號0-8的公式 我完全get 不到 只能叫gemini 給我提示 哎 需要再多練練


class Solution {
public:
    bool isValidSudoku(vector<vector<char>>& board) {
        
        // ==========================================
        //  坑點 1：陣列初始化 (Garbage Values)
        // ==========================================
        // 宣告在函式內的陣列必須加上 = {false} 進行聚合初始化。
        // 否則 C++ 記憶體會殘留隨機的垃圾值，導致判斷發瘋。
        // 這裡宣告了三個「防撞抽屜」，大小都是 9x9 (合法 Index 為 0~8)
        bool row_seen[9][9] = {false};
        bool column_seen[9][9] = {false};
        bool nine_grid_seen[9][9] = {false};

        // 雙層迴圈：完美的一次走訪 (One-Pass)，時間複雜度 O(1) (因為固定只跑 81 格)
        for(int i = 0; i < board.size(); i++){
            for(int j = 0; j < board[i].size(); j++){
                
                char temp = board[i][j];
                
                // 如果遇到空格 '.' 直接跳過，只有數字才需要檢查
                if(temp != '.'){
                    
                    // ==========================================
                    //  坑點 2：二維轉一維的「降維打擊」公式
                    // ==========================================
                    // i/3 算出目前在第幾個「大橫排」(R)
                    // j/3 算出目前在第幾個「大直排」(C)
                    // 套用公寓理論公式：R * 寬度(3) + C，精準算出 0~8 號九宮格編號
                    int nine_grid_num = ((i / 3) * 3) + (j / 3);
                    
                    // ==========================================
                    //  坑點 3：差一錯誤 (Off-By-One Error) 導致越界當機
                    // ==========================================
                    // 數獨的數字是 '1'~'9'，但我們的抽屜編號只到 0~8。
                    // 絕對不能用 temp - '0'！如果遇到數字 '9'，算出來是 9，
                    // 去開 [9] 號抽屜會直接引發記憶體區段錯誤 (Out of Bounds)！
                    // 必須減去 '1'，讓數字 '1' 變成 index 0，數字 '9' 變成 index 8。
                    int index = temp - '1';
                    
                    // 1. 檢查橫排抽屜：第 i 橫排的 index 數字出現過了嗎？
                    if(row_seen[i][index] == false){
                        row_seen[i][index] = true;  // 沒撞車，登記起來
                    }
                    else{
                        return false;               // 撞車！立刻判定無效
                    }
                    
                    // 2. 檢查直排抽屜：第 j 直排的 index 數字出現過了嗎？
                    if(column_seen[j][index] == false){
                        column_seen[j][index] = true;
                    }
                    else{
                        return false;
                    }
                    
                    // 3. 檢查九宮格抽屜：第 nine_grid_num 個九宮格的 index 數字出現過了嗎？
                    if(nine_grid_seen[nine_grid_num][index] == false){
                        nine_grid_seen[nine_grid_num][index] = true;
                    }
                    else{
                        return false;
                    }
                }
            }
        }
        
        // 掃描完 81 格都沒有任何撞車紀錄，代表這是一個合法的數獨棋盤！
        return true;
    }
};
