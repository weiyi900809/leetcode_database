
226. Invert Binary Tree

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 * int val;
 * TreeNode *left;
 * TreeNode *right;
 * TreeNode() : val(0), left(nullptr), right(nullptr) {}
 * TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 * TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* invertTree(TreeNode* root) {
        // 1. 邊界防禦：如果這是一塊空地 (空樹)，直接回傳空地，結束任務。
        if(root == nullptr){
            return root; // 這裡直接 return nullptr 也是一樣的意思
        }
        
        // 2. 核心動作：總裁親自把直屬的「左副總」跟「右副總」互換位置
        TreeNode* temp = root->left;
        root->left = root->right;
        root->right = temp;
        

        // 3. 向下交辦任務：
        // 總裁換完直屬部門後，下令讓左右副總各自去翻轉他們底下的部門
        invertTree(root->left);
        invertTree(root->right); // 【修正】這裡改成交辦給右部門
        
        // 任務全部完成，總裁回報整家翻轉後的公司結構
        return root;
    }
};

112. Path Sum

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 * int val;
 * TreeNode *left;
 * TreeNode *right;
 * TreeNode() : val(0), left(nullptr), right(nullptr) {}
 * TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 * TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool hasPathSum(TreeNode* root, int targetSum) {
        // ==========================================
        // 1. 嚴謹邊界防禦 (處理空樹或走到死胡同)
        // ==========================================
        // 如果總裁點到一塊空地 (根本沒有節點)，代表這條路不可能湊出任何預算。
        // 直接回報 false (失敗)。
        if(root == nullptr){
            return false;
        }

        // ==========================================
        // 2. 終止條件 (真正的基層員工 / 葉節點結算)
        // ==========================================
        // 檢查自己是不是「沒有左下屬也沒有右下屬」的最底層員工。
        if(root->left == nullptr && root->right == nullptr ){
            // 既然是最底層，就直接核對：「長官交辦下來的最後預算 (targetSum)，
            // 是不是剛好等於我自己的薪水 (root->val)？」
            // 等於就回傳 true，不等於就回傳 false。
            return targetSum == (root->val);
        }

        // ==========================================
        // 3. 遞迴交辦任務 (主管往下派發剩餘預算)
        // ==========================================
        // 既然程式能走到這裡，代表目前這個節點「絕對是個主管 (不是葉節點)」。
        // 主管扣掉自己的薪水後，把「剩下的預算」分別交辦給左部門和右部門去湊。
        bool left_side_path_ismeetsum = hasPathSum(root->left, targetSum - (root->val));
        bool right_side_path_ismeetsum = hasPathSum(root->right, targetSum - (root->val));
        
        // ==========================================
        // 4. 統整回報
        // ==========================================
        // 只要左邊或右邊「有任何一邊」成功回報達標 (true)，
        // 我這個主管的任務就算大功告成！(使用 || 邏輯或)
        return left_side_path_ismeetsum || right_side_path_ismeetsum;
    }
};
