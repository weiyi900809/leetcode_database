69. Sqrt(x)

class Solution {
public:
    int mySqrt(int x) {
        // ==========================================
        // 🛡️ 坑 1：邊界條件防呆 (Guard Clause)
        // ==========================================
        // 0 的平方根是 0，1 的平方根是 1。
        // 提早攔截，避免浪費時間進去跑二分搜尋。
        if( x == 1 || x == 0 ) return x;
        
        // ==========================================
        // 🚨 坑 2：虛擬陣列思維 (拒絕 Memory Limit Exceeded)
        // ==========================================
        // 絕對不能開 vector<int> 陣列！如果 x 是 21 億，開陣列會直接撐爆記憶體當機。
        // 我們直接把「數字本身」當作陣列的 Index (數線的概念)。
        int left = 1;
        int right = x;
        
        // 啟動二分搜尋 (終極密碼迴圈)
        // 當 left > right 時，代表整個區間都被切完了，迴圈才會停止。
        while (left <= right) {
            
            // ==========================================
            // 🚨 坑 3：防禦整數溢位 (Integer Overflow)
            // ==========================================
            // 寫法一：用 left + (right - left) / 2 可以防止 (left + right) 本身溢位。
            // 寫法二：必須宣告成 long long！因為等一下要做 mid * mid，
            // 如果 mid 很大 (例如 100 萬)，平方後會高達 10^12，直接塞爆 32-bit 的 int 導致當機。
            long long mid = left + (right - left) / 2;
            
            // 🎯 開始判斷 mid 的平方跟 x 的關係：
            
            // 情況 1：完美命中！mid 的平方剛好等於 x，提早下班。
            if( (mid * mid) == x ){
                return mid;
            }
            // 情況 2：猜得太小了！代表答案在右半邊，把 left 移到 mid 的右邊一格。
            else if( (mid * mid) < x ){ 
                left = mid + 1 ;
            }
            // 情況 3：猜得太大了！代表答案在左半邊，把 right 移到 mid 的左邊一格。
            else{
                right = mid - 1 ;
            }
        }
        
        // ==========================================
        // 🔑 坑 4：為什麼最後回傳 right？
        // ==========================================
        // 題目要求「無條件捨去小數」。
        // 當迴圈結束 (left > right) 且沒有完美命中時，
        // right 指標會剛好停在「平方值小於 x 的最大整數」上面！
        // (例如 x=8 時，最後 right 會停在 2，left 會跑到 3)。
        return right;
    }
};
