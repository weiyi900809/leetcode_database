836. Rectangle Overlap

class Solution {
public:
    bool isRectangleOverlap(vector<int>& rec1, vector<int>& rec2) {
        bool x_overlap = false;
        bool y_overlap = false;
        
        // ==========================================
        // 第一步：座標解構 (Unpacking)
        // ==========================================
        // 將長方形的座標提取出來，賦予直覺的變數名稱。
        // [x1, y1] 代表左下角，[x2, y2] 代表右上角。
        // 這個好習慣能大幅提升可讀性，避免程式碼充滿 rec1[0] 這類魔術數字。
        int ax1 = rec1[0], ay1 = rec1[1], ax2 = rec1[2], ay2 = rec1[3];
        int bx1 = rec2[0], by1 = rec2[1], bx2 = rec2[2], by2 = rec2[3];
        
        // ==========================================
        // 第二步：X 軸降維投影 (水平影子檢查)
        // ==========================================
        // 要在 1D 水平線上重疊，必須嚴格滿足「互相侵門踏戶」：
        // 1. A 的右邊 (ax2) 必須越過 B 的左邊 (bx1)
        // 2. B 的右邊 (bx2) 必須越過 A 的左邊 (ax1)
        // 兩者必須同時成立 (&&)。
        if(ax2 > bx1 && bx2 > ax1) {
            x_overlap = true;
        }
        
        // ==========================================
        // 第三步：Y 軸降維投影 (垂直影子檢查)
        // ==========================================
        // 邏輯同上，只不過換成上下邊界：
        // 1. A 的上面 (ay2) 必須越過 B 的下面 (by1)
        // 2. B 的上面 (by2) 必須越過 A 的下面 (ay1)
        if(ay2 > by1 && by2 > ay1) {
            y_overlap = true;
        }
        
        // ==========================================
        // 最終判定：2D 空間的重疊法則
        // ==========================================
        // 只有當 X 軸的影子有交集「且」Y 軸的影子也有交集時，
        // 這兩個 2D 長方形才算是真正重疊在一起。
        return y_overlap && x_overlap;
    }
};
