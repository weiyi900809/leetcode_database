200. Number of Islands


class Solution {
private:
    // ==========================================
    // DFS 偵察兵：負責把相連的陸地全部踩扁變成水
    // ==========================================
    void dfs(vector<vector<char>>& grid, int r, int c) {
        
        // ==========================================
        // 🚨 坑點 1：條件判斷的「絕對順序」(Short-circuit)
        // ==========================================
        // 你寫得非常完美：先檢查「有沒有越界」，再檢查「是不是海水」。
        // 如果把順序反過來，寫成 if(grid[r][c] == '0' || r < 0 ...)
        // 當 r = -1 時，程式會先去讀取 grid[-1][c]，直接觸發 Segmentation Fault 死當！
        // 記住：永遠要先確保座標合法，才能去讀取陣列裡面的值。
        
        // 1. 終止條件：越界防禦與海水撤退
        if(r < 0 || r >= grid.size()) return;     // 上下越界
        if(c < 0 || c >= grid[0].size()) return;  // 左右越界
        if(grid[r][c] == '0') return;             // 已經是水了，撤退
        
        // ==========================================
        // 🚨 坑點 2：無窮迴圈的深淵 (忘記沉島)
        // ==========================================
        // 如果漏寫了這一行，DFS 會在兩個相連的 '1' 之間無限來回互相呼叫，
        // 直到 Stack Overflow (堆疊溢位) 把記憶體撐爆。
        // Sinking Island (沉島) 策略是我們捨棄 visited 陣列的底氣。
        grid[r][c] = '0';
        
        // 3. 展開平行宇宙：往上下左右派出分身，繼續掃蕩相連的陸地
        dfs(grid, r-1, c); // 上
        dfs(grid, r+1, c); // 下
        dfs(grid, r, c-1); // 左
        dfs(grid, r, c+1); // 右
    }

public:
    int numIslands(vector<vector<char>>& grid) {
        
        // ==========================================
        // 🚨 坑點 3：隱形的空地圖炸彈
        // ==========================================
        // 如果沒有這道防線，當傳入空陣列 [] 時，
        // 底下的 grid[0].size() 會因為找不到第 0 列而直接崩潰。
        // 這是資深工程師必備的「防禦性編程」。
        if (grid.empty()) return 0;
        
        int islands = 0;
        int rows = grid.size();
        int cols = grid[0].size();
        
        // 主帥親征：雙層迴圈掃描地圖
        for (int r = 0; r < rows; r++) {
            for (int c = 0; c < cols; c++) {
                
                // 只要發現一塊沒被踩扁的陸地
                if (grid[r][c] == '1') {
                    islands++;           // 記錄發現新島嶼
                    dfs(grid, r, c);     // 派出 DFS，把這座島連根拔起沉入海底
                }
                
            }
        }
        
        return islands;
    }
};
