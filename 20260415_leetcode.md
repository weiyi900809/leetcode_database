202. Happy Number

class Solution {
public:
    int SumOfSquare (int input){
        int sum = 0;
        while(input > 0){
            // 把餘數抓出來自己乘自己，完美避開 ^ 的地雷
            sum += (input % 10) * (input % 10); 
            input /= 10;
        }
        return sum;
    }

    bool isHappy(int n) {
        unordered_set<int> num_seen;

        // 當 n 還不是 1 (還沒快樂)，而且 n 還沒出現在黑名單時 (還沒死循環)，就繼續跑
        while (n != 1 && num_seen.count(n) == 0) {
            
            // 1. 先把現在這個 n 拍照存證，加入黑名單
            num_seen.insert(n);
            
            // 2. 讓 n 變身成下一個數字！
            n = SumOfSquare(n);
        }

        // 迴圈會停下來，只有兩種可能：
        // 1. n 真的變成 1 了 (跳出迴圈)
        // 2. n 踩到黑名單了，無限迴圈了 (跳出迴圈)
        // 所以我們只要檢查最後的 n 是不是 1 就好！
        return n == 1; 
    }
};
